1"""PM-sync pipeline — direction → tracker issue. 

2 

3For each pending direction in ``apps/<app>/directions/``: 

4 

51. Parse the directory → ``Direction`` record. 

62. Fast pre-check via ``backpressure.validator.validate_direction``. 

73. Insufficient → ``record_needs_direction`` + status = ``needs-direction``. 

84. Sufficient → invoke PM persona via ``text_run`` with the direction body + 

9 the canonical context prelude (loader from Phase 0). 

105. Persist the PM JSON response under ``pm_result`` in ``state.yaml``. 

116. ``open_or_update_tracker_issue`` (idempotent). 

127. Status → ``pm-validated`` (Phase 2 spawns child story issues from 

13 ``pm_result.child_stories``). 

14 

15``dry_run=True`` is a PURE PREVIEW — it skips the LLM and GitHub calls AND 

16makes no persistent mutation: no ``state.yaml`` write (status/pm_result stay 

17as-is), no ``runs`` row, no StoryRecord persisted, no GC close on disk. It 

18only computes what the real run WOULD decide and returns it in the summary 

19(``pm_result`` from a deterministic fixture, ``gc_closed`` as the would-close 

20list). This makes dry-run repeatable and, crucially, incapable of spawning 

21live dispatchable work: a "safe" preview that flipped a direction to 

22``pm-validated`` and persisted rebuild-stories once caused the 2026-07-20 

23self-tick incident. The real (non-dry) run is the only writer. 

24"""

138def _build_pm_prompt(direction: Direction, context_prelude: str) -> str: 

139 """Construct the user-message prompt for the PM persona text_run.""" 

140 flow_text = "" 

141 api_text = "" 

142 if direction.has_flow: 

143 try: 

144 flow_text = (direction.dir_path / "flow.md").read_text(encoding="utf-8") 

145 except FileNotFoundError: 

146 flow_text = "" 

147 if direction.has_api_spec: 

148 try: 

149 api_text = (direction.dir_path / "api_spec.md").read_text(encoding="utf-8") 

150 except FileNotFoundError: 

151 api_text = "" 

152 

153 parts: list[str] = [] 

154 parts.append(context_prelude.rstrip()) 

155 parts.append("\n---\n# Direction\n") 

156 parts.append(f"id: {direction.id}\nslug: {direction.slug}\napp: {direction.app}\n") 

157 parts.append(f"frontmatter: {json.dumps(direction.raw_frontmatter, default=str)}\n") 

158 parts.append("\n## direction.md\n") 

159 parts.append(direction.raw_body.rstrip()) 

160 if flow_text: 

161 parts.append("\n\n## flow.md\n") 

162 parts.append(flow_text.rstrip()) 

163 if api_text: 

164 parts.append("\n\n## api_spec.md\n") 

165 parts.append(api_text.rstrip()) 

166 if direction.artifacts_paths: 

167 parts.append("\n\n## artifacts/ (filenames only)\n") 

168 for p in direction.artifacts_paths: 

169 parts.append(f"- {p.name}") 

170 return "\n".join(parts)

173def _dry_run_pm_result(direction: Direction, validation: ValidationResult) -> dict[str, Any]: 

174 """Deterministic stub PM result for dry-run mode. 

175 

176 Mirrors the structural validation: if backpressure is sufficient, emits a 

177 plausible "validated" record; otherwise emits a "needs-direction" record 

178 with the validator's missing list. No LLM call; no randomness. 

179 """ 

180 typ = direction.type_tag or "feature" 

181 priority_default = "p1" if typ in {"security", "bug"} else "p2" 

182 priority = direction.raw_frontmatter.get("priority") or priority_default 

183 

184 title_short = direction.title[:60] 

185 tracker_title = f"[DIRECTION] {title_short}" 

186 if len(tracker_title) > 69: 

187 tracker_title = tracker_title[:66] + "..." 

188 

189 if validation.is_valid: 

190 child_stories: list[dict[str, str]] = [] 

191 # Docs-typed directions get a single ``chain_kind: "docs"`` story by 

192 # default — the dry-run fixture mirrors what a live PM call would 

193 # emit for a context-bootstrap or onboarder-style direction. The 

194 # docs path skips test_design/test_impl/dev entirely. 

195 if typ == "docs": 

196 child_stories.append( 

197 { 

198 "title": title_short, 

199 "scope": "docs", 

200 "chain_kind": "docs", 

201 "rationale": "Documentation-only direction; routed through the docs chain.", 

202 "points": 2, 

203 } 

204 ) 

205 else: 

206 if direction.has_api_spec: 

207 child_stories.append( 

208 { 

209 "title": f"Implement API: {title_short}", 

210 "scope": "backend", 

211 "chain_kind": "tdd", 

212 "rationale": "Direction declares an API contract; one backend story to implement it.", 

213 "points": 3, 

214 } 

215 ) 

216 if direction.has_flow: 

217 child_stories.append( 

218 { 

219 "title": f"Implement UI flow: {title_short}", 

220 "scope": "frontend", 

221 "chain_kind": "tdd", 

222 "rationale": "Direction declares a user flow; one frontend story to implement it.", 

223 "points": 3, 

224 } 

225 ) 

226 if not child_stories: 

227 child_stories.append( 

228 { 

229 "title": title_short, 

230 "scope": "backend", 

231 "chain_kind": "tdd", 

232 "rationale": "Explore-tagged direction; single backend story as a starting point.", 

233 "points": 3, 

234 } 

235 ) 

236 return { 

237 "type": typ, 

238 "priority": priority, 

239 "has_sufficient_backpressure": True, 

240 "missing": [], 

241 "tracker_title": tracker_title, 

242 "tracker_body": ( 

243 f"**{direction.title}**\n\n{(direction.why or '_(no why captured)_').strip()}\n" 

244 ), 

245 "child_stories": child_stories, 

246 "labels": [typ, f"priority/{priority}"], 

247 "confidence": 0.75, 

248 } 

249 return { 

250 "type": typ, 

251 "priority": priority, 

252 "has_sufficient_backpressure": False, 

253 "missing": validation.missing, 

254 "tracker_title": tracker_title, 

255 "tracker_body": ( 

256 f"**{direction.title}**\n\n" 

257 f"_(needs more backpressure: {', '.join(validation.missing)})_\n" 

258 ), 

259 "child_stories": [], 

260 "labels": [typ, f"priority/{priority}", "needs-direction"], 

261 "confidence": 0.4, 

262 }

265# --------------------------------------------------------------------------- # 

266# Story sizing — enforce the PM persona's hard size rule 

267# --------------------------------------------------------------------------- # 

268 

269 

270# Per-story size ceilings the chain rejects on. Aligned with the PM persona 

271# prompt's hard sizing rule; any single child_story exceeding these gets sent 

272# back to the PM for further decomposition. Picked from observed dev-pass 

273# budgets: dev sandbox has a 600-iteration default with up to 10 chain 

274# retries, so a 200-iteration single-story estimate leaves ample headroom. 

275MAX_NEW_FILES_PER_STORY = 5 

276MAX_MODIFIED_FILES_PER_STORY = 2 

277MAX_SANDBOX_ITERATIONS_PER_STORY = 200 

278 

279# Cap on how many times the chain will re-prompt the PM with size feedback 

280# before accepting whatever it returns. Each re-prompt is a fresh LLM call; 

281# small bound keeps cost predictable without giving up after one try. 

282MAX_PM_REDECOMPOSITION_RETRIES = 3

285def _story_size_violations(child_story: dict[str, Any]) -> list[str]: 

286 """Return a human-readable list of size violations for one child_story. 

287 

288 Empty list means the story fits. Missing/None estimates are treated as 

289 zero (assumed-fine) but logged via the returned ``"missing_estimate_..."`` 

290 sentinel so the operator can spot under-instrumented PM output. 

291 """ 

292 violations: list[str] = [] 

293 nf = child_story.get("estimated_new_files") 

294 mf = child_story.get("estimated_modified_files") 

295 it = child_story.get("estimated_sandbox_iterations") 

296 if nf is None: 

297 violations.append("missing_estimate_new_files") 

298 elif isinstance(nf, int) and nf > MAX_NEW_FILES_PER_STORY: 

299 violations.append( 

300 f"estimated_new_files={nf} exceeds max {MAX_NEW_FILES_PER_STORY}" 

301 ) 

302 if mf is None: 

303 violations.append("missing_estimate_modified_files") 

304 elif isinstance(mf, int) and mf > MAX_MODIFIED_FILES_PER_STORY: 

305 violations.append( 

306 f"estimated_modified_files={mf} exceeds max {MAX_MODIFIED_FILES_PER_STORY}" 

307 ) 

308 if it is None: 

309 violations.append("missing_estimate_sandbox_iterations") 

310 elif isinstance(it, int) and it > MAX_SANDBOX_ITERATIONS_PER_STORY: 

311 violations.append( 

312 f"estimated_sandbox_iterations={it} exceeds max " 

313 f"{MAX_SANDBOX_ITERATIONS_PER_STORY}" 

314 ) 

315 return violations

318def _validate_pm_story_sizes(pm_result: dict[str, Any]) -> dict[int, list[str]]: 

319 """Return ``{story_index: [violations]}`` for stories that exceed limits. 

320 

321 Skips when ``has_sufficient_backpressure`` is False (no stories spawn 

322 anyway) or ``child_stories`` is empty. Missing-estimate sentinels are 

323 included so the operator can see when PM is emitting unsized output. 

324 """ 

325 if not pm_result.get("has_sufficient_backpressure"): 

326 return {} 

327 out: dict[int, list[str]] = {} 

328 for idx, story in enumerate(pm_result.get("child_stories") or []): 

329 violations = _story_size_violations(story) 

330 # Treat the "missing_estimate_*" sentinels as soft warnings — they 

331 # don't trigger re-prompts on their own (back-compat with PMs that 

332 # haven't been updated to emit the fields). Only real-exceed 

333 # violations gate the re-prompt loop. 

334 hard = [v for v in violations if not v.startswith("missing_estimate_")] 

335 if hard: 

336 out[idx] = hard 

337 return out

340def _format_redecomposition_feedback(violations: dict[int, list[str]]) -> str: 

341 """Compose the operator-visible feedback string the chain hands back to PM.""" 

342 lines = [ 

343 "## Chain feedback — story sizes exceed dev's per-pass budget", 

344 "", 

345 "Your previous decomposition included stories that are too large for a", 

346 "single dev sandbox pass. The chain rejects oversized stories so dev", 

347 "doesn't burn retry budget on un-completable slices. Re-decompose the", 

348 "flagged stories into smaller vertical slices and re-emit the full PM", 

349 "JSON.", 

350 "", 

351 "Per-story size ceilings (HARD):", 

352 f" - estimated_new_files ≤ {MAX_NEW_FILES_PER_STORY}", 

353 f" - estimated_modified_files ≤ {MAX_MODIFIED_FILES_PER_STORY}", 

354 f" - estimated_sandbox_iterations ≤ {MAX_SANDBOX_ITERATIONS_PER_STORY}", 

355 "", 

356 "Flagged stories from your last output:", 

357 ] 

358 for idx, vs in sorted(violations.items()): 

359 lines.append(f" - story[{idx}]: {'; '.join(vs)}") 

360 lines.append("") 

361 lines.append( 

362 "Re-emit the full PM JSON with every child_story now within the limits." 

363 ) 

364 return "\n".join(lines)

367def _call_pm_persona( 

368 direction: Direction, 

369 app_repo_path: Path, 

370 software_factory_root: Path, 

371) -> dict[str, Any]: 

372 """Real LLM call. Returns the parsed PM JSON result. 

373 

374 Loops up to ``MAX_PM_REDECOMPOSITION_RETRIES`` times when the returned 

375 ``child_stories`` exceed the chain's per-story size ceilings, threading 

376 structured feedback through to each retry so the PM can correct. 

377 """ 

378 # Import lazily so dry-run paths don't pull litellm. 

379 from factory.runner import text_run 

380 

381 persona = "pm" 

382 persona_prompt = _read_persona_prompt(persona) 

383 

384 chain = _resolve_chain_for_direction(direction, software_factory_root) 

385 context_prelude = compose_context_prelude( 

386 persona=persona, 

387 app_repo_path=app_repo_path, 

388 direction_chain=chain, 

389 software_factory_root=software_factory_root, 

390 ) 

391 direction_block = _build_pm_prompt(direction, context_prelude) 

392 

393 base_prompt = ( 

394 f"{persona_prompt.rstrip()}\n\n" 

395 "---\n\n" 

396 "## Input\n\n" 

397 f"{direction_block}\n\n" 

398 "---\n\n" 

399 "Return the JSON object for this direction. No prose outside the JSON." 

400 ) 

401 model_id = route(persona) 

402 

403 feedback: str | None = None 

404 last_result: dict[str, Any] | None = None 

405 for attempt in range(1, MAX_PM_REDECOMPOSITION_RETRIES + 2): 

406 # On retries, prepend the chain's structured feedback so the PM sees 

407 # what failed and re-decomposes accordingly. The persona prompt's 

408 # sizing rule already covers the policy; this just surfaces which 

409 # specific stories tripped it. 

410 full_prompt = base_prompt 

411 if feedback is not None: 

412 full_prompt = f"{base_prompt}\n\n---\n\n{feedback}\n" 

413 

414 # Cap output tokens at 2048 — PM JSON is small. 

415 result = text_run( 

416 persona=persona, 

417 prompt=full_prompt, 

418 model_id=model_id, 

419 schema=_PM_SCHEMA, 

420 max_tokens=2048, 

421 ) 

422 if not isinstance(result, dict): 

423 # text_run only returns str when schema is None; treat anything 

424 # else as failure. 

425 raise RuntimeError("PM text_run returned a non-dict for schema-mode call") 

426 

427 last_result = result 

428 violations = _validate_pm_story_sizes(result) 

429 if not violations: 

430 return result 

431 if attempt > MAX_PM_REDECOMPOSITION_RETRIES: 

432 break 

433 feedback = _format_redecomposition_feedback(violations) 

434 

435 # Out of retries — return the last result with the violations recorded 

436 # so downstream code can see we tried. Operators surface these via 

437 # ``factory why`` / the tracker body. 

438 assert last_result is not None 

439 last_result.setdefault("_chain_warnings", []).append( 

440 { 

441 "kind": "story_sizes_exceeded_after_retries", 

442 "retries_used": MAX_PM_REDECOMPOSITION_RETRIES, 

443 "violations": _validate_pm_story_sizes(last_result), 

444 } 

445 ) 

446 return last_result

466def pm_sync( 

467 app: str, 

468 software_factory_root: Path, 

469 *, 

470 dry_run: bool = False, 

471 github_client: Any = None, 

472 state_db_path: Path | None = None, 

473 pending_statuses: frozenset[str] = frozenset({"created", "needs-direction"}), 

474 run_gc: bool = True, 

475) -> PMSyncSummary: 

476 """Run the PM-sync pipeline once for ``app``. Returns a summary record. 

477 

478 ``pending_statuses`` narrows which pending directions are processed. 

479 The default (operator-invoked ``factory pm-sync``) re-validates 

480 ``needs-direction`` entries too, so an operator who just fleshed out a 

481 direction gets it re-checked. Automated callers should pass 

482 ``frozenset({"created"})`` — re-validating an unchanged insufficient 

483 direction on every tick only re-posts the same tracker-issue comment. 

484 

485 ``run_gc=False`` skips this call's own stale-scheduled-direction GC pass 

486 (see ``factory.directions.gc``). ``maybe_auto_pm_sync`` runs GC itself 

487 independently of the ``created``-gate that decides whether it calls into 

488 this function at all, then passes ``run_gc=False`` here to avoid running 

489 the GC pass twice in the same tick. Manual ``factory pm-sync`` callers 

490 should leave this at the default ``True``. 

491 """ 

492 root = Path(software_factory_root) 

493 db_path = state_db_path or (root / "state" / "factory.db") 

494 

495 app_config: AppConfig | None = None 

496 try: 

497 app_config = load_app_config(app, root) 

498 except FileNotFoundError as exc: 

499 # Hard error — without a config we don't know the GitHub repo. 

500 # Dry-run still works (no GH calls), but real-run cannot. 

501 if not dry_run: 

502 raise RuntimeError(f"Cannot run pm-sync without app config: {exc}") from exc 

503 

504 # In real-run, the GitHub client is required. 

505 if not dry_run and github_client is None: 

506 raise RuntimeError( 

507 "github_client is required for real pm-sync; pass --dry-run for offline use" 

508 ) 

509 

510 # Self-tick guard (Tier 3 — FACTORY-SELF-TICK). The factory building its OWN 

511 # repo (apps/factory) is OFF by default: while ``self_tick_enabled`` is 

512 # False we do NOT turn apps/factory directions into chain stories, so merely 

513 # bootstrapping apps/factory (config + directions on disk) never silently 

514 # starts the factory ticking on itself. The orchestrator flips the flag to 

515 # enable self-improvement deliberately. Non-factory apps are unaffected 

516 # (the guard only triggers for the factory's own repo). The chain-side 

517 # staging gate is a SEPARATE, always-on protection — this guard only decides 

518 # whether self-improvement work enters the chain at all. 

519 if ( 

520 app_config is not None 

521 and targets_factory_repo(app_config.repo) 

522 and not app_config.self_tick_enabled 

523 ): 

524 return PMSyncSummary() 

525 

526 summary = PMSyncSummary() 

527 pending = [ 

528 d for d in pending_directions(app, root, db_path) if d.status in pending_statuses 

529 ] 

530 summary.processed = len(pending) 

531 

532 # App repo path for the context prelude. Phase 7 resolves this via the 

533 # app's ``config.yaml::app_repo_path`` (default ``../<name>``); on 

534 # operator-typical layouts that's a sibling of the factory root. 

535 # If the path doesn't exist on disk yet, the loader returns the 

536 # NO_CONTEXT_AVAILABLE notice cleanly. 

537 # ``load_app_config`` may raise FileNotFoundError when the apps/ entry 

538 # for this app doesn't exist (e.g. test fixtures without a config) — 

539 # the earlier ``app_config = load_app_config(...)`` block has already 

540 # handled that case, so reuse the loaded record when available. 

541 if app_config is not None: 

542 app_repo_path = resolve_app_repo_path(app_config, root) 

543 else: 

544 # Dry-run path with no on-disk config: synthesize a stub. 

545 app_repo_path = root / "apps" / app 

546 

547 for direction in pending: 

548 try: 

549 validation = validate_direction(direction) 

550 

551 if not validation.is_valid: 

552 # Dry-run is a PURE PREVIEW: it must not mutate any persistent 

553 # state (no state.yaml write, no runs row, no GitHub call). A 

554 # dry-run that flips a direction's status or records a pm row 

555 # is the proxy!=real footgun that once let a "safe" preview 

556 # spawn live rebuild-stories (2026-07-20 self-tick incident). 

557 # We still compute pm_result so the returned summary reflects 

558 # exactly what the real run would decide. 

559 pm_result = _dry_run_pm_result(direction, validation) 

560 if not dry_run: 

561 assert app_config is not None and github_client is not None 

562 record_needs_direction( 

563 direction, 

564 validation.missing, 

565 app_config, 

566 github_client, 

567 pm_result=pm_result, 

568 ) 

569 merge_state( 

570 direction, 

571 {"pm_result": pm_result, "validation_issues": validation.issues}, 

572 ) 

573 mark_direction_status( 

574 direction, 

575 "needs-direction", 

576 by="factory.chain.pm_sync", 

577 details={"missing": validation.missing}, 

578 ) 

579 summary.needs_direction += 1 

580 continue 

581 

582 # Backpressure is sufficient — invoke PM persona (or fixture). 

583 # Dry-run stays a pure preview: no PM LLM call, no state.yaml 

584 # write, no runs row, no GitHub issue — it only computes what the 

585 # real run WOULD do and surfaces it in the returned summary. 

586 if dry_run: 

587 pm_result = _dry_run_pm_result(direction, validation) 

588 else: 

589 pm_result = _call_pm_persona(direction, app_repo_path, root) 

590 merge_state(direction, {"pm_result": pm_result}) 

591 assert app_config is not None and github_client is not None 

592 open_or_update_tracker_issue( 

593 direction, 

594 app_config, 

595 github_client, 

596 pm_result=pm_result, 

597 software_factory_root=root, 

598 ) 

599 mark_direction_status( 

600 direction, 

601 "pm-validated", 

602 by="factory.chain.pm_sync", 

603 details={"confidence": pm_result.get("confidence")}, 

604 ) 

605 summary.validated += 1 

606 

607 # Spawn StoryRecord rows for each child_story in pm_result. 

608 # In dry-run, no GitHub issue is created; story_file_path uses 

609 # a placeholder issue number of 0. In real-run, the chain 

610 # opens the GH issue and uses the real number. 

611 try: 

612 from factory.chain.handlers import handle_stories_spawned 

613 

614 if app_config is None and not dry_run: 

615 raise RuntimeError("app config required for real-run spawn") 

616 # In dry-run we still need an AppConfig stub for the handler 

617 # signature; build a minimal one from the app name if missing. 

618 spawn_config: AppConfig 

619 if app_config is not None: 

620 spawn_config = app_config 

621 else: 

622 spawn_config = AppConfig(name=app, repo="<dry-run>") 

623 handle_stories_spawned( 

624 direction=direction, 

625 pm_result=pm_result, 

626 app_config=spawn_config, 

627 software_factory_root=root, 

628 dry_run=dry_run, 

629 db_path=db_path, 

630 github_client=github_client, 

631 ) 

632 except Exception as spawn_exc: 

633 # Don't fail pm-sync on a spawn error; record it. 

634 summary.errors.append( 

635 (direction.id or direction.slug, f"story-spawn: {spawn_exc!r}") 

636 ) 

637 

638 except Exception as exc: # pragma: no cover - exercised in error tests 

639 summary.errors.append((direction.id or direction.slug, repr(exc))) 

640 

641 # GC pass: close scheduler-filed directions that have sat unactioned at 

642 # needs-direction past the threshold (audit 2026-07-18, leak 2 of 4). 

643 # Best-effort — a GC failure must never fail the pm-sync pass it rides 

644 # along with. Skipped when the caller (``maybe_auto_pm_sync``) already 

645 # ran GC itself this tick — see ``run_gc``'s docstring. 

646 if run_gc: 

647 try: 

648 from factory.directions.gc import gc_stale_scheduled_directions 

649 

650 summary.gc_closed = gc_stale_scheduled_directions( 

651 app, 

652 root, 

653 app_config, 

654 github_client, 

655 dry_run=dry_run, 

656 ) 

657 except Exception as gc_exc: # noqa: BLE001 - GC is a side pass, never fatal 

658 summary.errors.append(("__gc__", repr(gc_exc))) 

659 

660 return summary

663def _pm_runs_last_hour(db_path: Path) -> int: 

664 """Count real ``pm`` persona rows recorded in the last hour.""" 

665 import sqlite3 

666 from datetime import UTC, datetime, timedelta 

667 

668 if not db_path.exists(): 

669 return 0 

670 cutoff = (datetime.now(UTC) - timedelta(hours=1)).isoformat() 

671 try: 

672 conn = sqlite3.connect(str(db_path)) 

673 try: 

674 row = conn.execute( 

675 "SELECT COUNT(*) FROM runs WHERE persona='pm' AND ts > ?", 

676 (cutoff,), 

677 ).fetchone() 

678 return int(row[0]) if row else 0 

679 finally: 

680 conn.close() 

681 except sqlite3.OperationalError: 

682 # Fresh DB without a runs table yet — nothing has run. 

683 return 0

686def maybe_auto_pm_sync( 

687 app: str, 

688 software_factory_root: Path, 

689 *, 

690 dry_run: bool = False, 

691 github_client_factory: Any = None, 

692 state_db_path: Path | None = None, 

693) -> tuple[PMSyncSummary | None, str]: 

694 """Tick-driven auto-triage of pending directions. 

695 

696 Directions filed by the scheduled personas (or ``factory tell``) used to 

697 sit in ``status: created`` until an operator remembered to run 

698 ``factory pm-sync`` — generated work rotted at the inbox. When 

699 ``auto_pm_sync.enabled`` is set, every tick calls this; it runs the 

700 pm-sync pipeline only when there is something to triage and the 

701 ``pm_invocations_per_hour`` budget (measured from real ``pm`` rows in the 

702 runs table) has headroom, so an erroring direction can't burn spend by 

703 being retriaged on every tick. 

704 

705 ``github_client_factory`` is called (no args) only when a real sync will 

706 actually run, so ticks on hosts without GitHub credentials don't fail 

707 when there's nothing to triage. 

708 

709 Only ``status: created`` directions are auto-triaged. ``needs-direction`` 

710 entries are deliberately excluded from the LLM-heavy pm-sync pipeline: 

711 they failed backpressure validation and re-validating them unchanged 

712 every tick just re-posts the same "Needs direction" comment on their 

713 tracker issues (observed live 2026-06-11: 15 stuck directions x one 

714 comment per 5-minute tick). They are re-checked by an operator-invoked 

715 ``factory pm-sync`` after the missing artifacts are added. 

716 

717 The stale-scheduled-direction GC pass (``factory.directions.gc``) is the 

718 one exception: it runs on every tick regardless of whether a ``created`` 

719 direction is pending, BEFORE the ``created``-gate below. GC is pure 

720 filesystem work (plus an optional best-effort GitHub issue-close) — no 

721 LLM call — so it doesn't need to wait behind the same gate that protects 

722 the expensive pm-sync pipeline. Previously GC only rode along inside 

723 ``pm_sync()``, which this function only reached when a ``created`` 

724 direction existed; a backlog of stale ``needs-direction`` directions 

725 with nothing fresh alongside meant GC never fired on the automated tick 

726 (audit 2026-07-18). 

727 

728 Returns ``(summary_or_None, reason)`` with reason in 

729 {"disabled", "no_pending", "rate_limited", "synced"}. 

730 """ 

731 from datetime import UTC, datetime 

732 

733 from factory.directions.gc import gc_stale_scheduled_directions, is_gc_eligible 

734 from factory.settings.loader import load_settings 

735 

736 root = Path(software_factory_root) 

737 db_path = state_db_path or (root / "state" / "factory.db") 

738 

739 settings = load_settings(root) 

740 if not settings.auto_pm_sync.enabled: 

741 return None, "disabled" 

742 

743 pending = pending_directions(app, root, db_path) 

744 

745 # GC pass — independent of the ``created``-gate, see docstring above. 

746 # Only bothers loading the app config / GitHub client when at least one 

747 # direction actually crosses the GC threshold, so a tick with nothing 

748 # pending (or nothing GC-eligible) still never touches GitHub — hosts 

749 # without credentials must not fail on an idle queue. 

750 now = datetime.now(UTC) 

751 github_client: Any = None 

752 if any(is_gc_eligible(d, now=now) for d in pending): 

753 gc_app_config = None 

754 if not dry_run: 

755 try: 

756 gc_app_config = load_app_config(app, root) 

757 except FileNotFoundError: 

758 gc_app_config = None 

759 if github_client_factory is not None: 

760 github_client = github_client_factory() 

761 try: 

762 gc_stale_scheduled_directions( 

763 app, 

764 root, 

765 gc_app_config, 

766 github_client, 

767 dry_run=dry_run, 

768 now=now, 

769 ) 

770 except Exception: # noqa: BLE001 - GC is a side pass, never fatal 

771 pass 

772 

773 auto_statuses = frozenset({"created"}) 

774 if not any(d.status in auto_statuses for d in pending): 

775 return None, "no_pending" 

776 

777 if _pm_runs_last_hour(db_path) >= settings.rate_limits.pm_invocations_per_hour: 

778 return None, "rate_limited" 

779 

780 if github_client is None and not dry_run and github_client_factory is not None: 

781 github_client = github_client_factory() 

782 

783 summary = pm_sync( 

784 app=app, 

785 software_factory_root=root, 

786 dry_run=dry_run, 

787 github_client=github_client, 

788 state_db_path=db_path, 

789 pending_statuses=auto_statuses, 

790 run_gc=False, 

791 ) 

792 return summary, "synced"
