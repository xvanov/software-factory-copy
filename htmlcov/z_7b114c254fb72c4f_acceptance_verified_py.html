1"""Gate: ``acceptance-verified`` (WS1.2 — independent acceptance oracle). 

2 

3The problem this closes 

4======================= 

5 

6Loop-4 makes the dev author AND run its own tests. That is great for 

7convergence, but a coder that writes the tests judging it can reward-hack — 

8special-case the exact assertions, weaken them, or delete the hard ones 

9(ImpossibleBench: hiding the acceptance tests from the coder drops cheating to 

10~0). Every other merge gate re-derives truth from artifacts the dev produced 

11(``tests-green`` re-runs the dev's suite; ``tests-meaningful`` scans the dev's 

12tests). None of them is INDEPENDENT of the dev. 

13 

14This gate is that independent layer. It runs an acceptance test authored from 

15the direction's acceptance criteria — the SPEC ONLY, blind to the dev's code 

16and the dev's tests — that the dev never sees or edits (authored at story spawn, 

17stored under ``state/acceptance/<app>/<story_id>/`` OUTSIDE the dev worktree; 

18see ``factory.chain.acceptance``). At merge time this gate copies that test into 

19the merge-candidate checkout, runs it against the app's python env, and passes 

20iff it exits 0 — so a story whose OWN tests are green but whose behaviour 

21violates an acceptance criterion is caught here. 

22 

23The blocking decision keys off ``story.acceptance_expected`` — set at spawn to 

24"app opted in AND the direction has ACs", INDEPENDENT of whether authoring 

25succeeded — NOT off the presence of the stored file. That is the fix for the 

26false-green hole: an authoring failure leaves ``acceptance_expected=True`` with 

27no stored test, and this gate then BLOCKS AUTHORITATIVELY (never a silent skip). 

28It is not a permanent dead-end: ``acceptance.reauthor_missing_oracles`` re-authors 

29expected-but-missing oracles on a later tick (spec-only, so still dev-blind), so 

30the story heals and gets gated for real. 

31 

32Resolution 

33========== 

34 

35* Not opted in (``gates.acceptance_oracle`` False): PASS (skip). Never required. 

36* Opted in but this story is NOT expected to have an oracle (no ACs): PASS 

37 (skip, not applicable). ``required_gate_labels`` does not require it. 

38* Expected + stored test readable: 

39 - real-run (checkout present): copy in, run, pass iff exit 0, remove. 

40 - dry-run / no checkout: NON-AUTHORITATIVE (cannot verify), never a pass. 

41* Expected + stored test MISSING/unreadable (authoring flaked): AUTHORITATIVE 

42 BLOCK (passed=False, authoritative=True) — self-heal re-authors next tick. 

43""" 

44 

45from __future__ import annotations 

46 

47import shutil 

48from pathlib import Path 

49 

50from factory.app_config import AppConfig 

51from factory.chain.acceptance import ref_is_readable 

52from factory.chain.gates.evaluator import GateResult, PRContext, _run_command 

53 

54_DEFAULT_ACCEPTANCE_COMMAND = "python -m pytest {test_file} -q" 

55 

56 

57def evaluate(pr: PRContext, app_config: AppConfig) -> GateResult: 

58 label = "acceptance-verified" 

59 gates = app_config.gates 

60 

61 # Not opted in: skip (pass). Mirrors the optional command gates — a missing 

62 # capability means "this gate does not apply", not "this gate fails". 

63 if not gates.acceptance_oracle: 

64 return GateResult( 

65 label=label, 

66 passed=True, 

67 reason="acceptance oracle not enabled for this app (skipped)", 

68 details={"acceptance_oracle": False}, 

69 ) 

70 

71 story = pr.story 

72 ref = getattr(story, "acceptance_test_ref", None) if story is not None else None 

73 # Expected is the required/blocking source of truth. Fall back to "a ref 

74 # exists" for legacy stories spawned before ``acceptance_expected`` existed. 

75 expected = bool(getattr(story, "acceptance_expected", False)) or bool(ref) 

76 

77 if not expected: 

78 # Opted in, but this story has no acceptance criteria to verify — not 

79 # applicable, and required_gate_labels does not require it. 

80 return GateResult( 

81 label=label, 

82 passed=True, 

83 reason="story has no acceptance criteria (not applicable, skipped)", 

84 details={"acceptance_oracle": True, "acceptance_expected": False}, 

85 ) 

86 

87 root = pr.software_factory_root 

88 

89 # Expected but the stored oracle is missing/unreadable → authoring flaked. 

90 # BLOCK AUTHORITATIVELY (never a silent pass). The tick self-heal 

91 # (reauthor_missing_oracles) re-authors it before a later merge attempt, so 

92 # this is not a permanent dead-end. 

93 if not ref_is_readable(story, root): 

94 return GateResult( 

95 label=label, 

96 passed=False, 

97 reason=( 

98 "acceptance oracle EXPECTED but not available " 

99 f"(ref={ref!r}, root={'set' if root else 'unset'}) — authoring " 

100 "failed; blocking until it is re-authored (self-heals next tick)" 

101 ), 

102 details={"authoritative": True, "acceptance_expected": True, 

103 "acceptance_test_ref": ref}, 

104 ) 

105 

106 # Need a real checkout to run against. Dry-run (no worktree) cannot 

107 # re-derive truth — never claim a merge-authoritative pass. 

108 if pr.dry_run or pr.repo_root is None: 

109 return GateResult( 

110 label=label, 

111 passed=False, 

112 reason="[dry-run] acceptance oracle present but not run (no checkout)", 

113 details={"authoritative": False, "acceptance_test_ref": ref}, 

114 ) 

115 

116 # Real-run: copy the independent test into the merge-candidate checkout, 

117 # run it against the app's env, then remove it (never leave it behind to be 

118 # committed). Named distinctively (story id, else head_sha) so it cannot 

119 # collide with app tests. 

120 # ref_is_readable(story, root) returned True above → story, ref, root all set. 

121 assert story is not None and ref is not None and root is not None 

122 sid = story.id if story.id is not None else pr.head_sha 

123 dest_name = f"test_acceptance_oracle_{sid}.py" 

124 dest = Path(pr.repo_root) / dest_name 

125 

126 p = Path(ref) 

127 stored = p if p.is_absolute() else Path(root) / p 

128 

129 cmd_template = gates.acceptance_test_command or _DEFAULT_ACCEPTANCE_COMMAND 

130 cmd = cmd_template.format(test_file=dest_name) 

131 try: 

132 shutil.copyfile(stored, dest) 

133 exit_code, output = _run_command(cmd, cwd=Path(pr.repo_root)) 

134 finally: 

135 try: 

136 if dest.exists(): 

137 dest.unlink() 

138 except OSError: 

139 pass 

140 

141 return GateResult( 

142 label=label, 

143 passed=exit_code == 0, 

144 reason=f"ran independent acceptance oracle exit_code={exit_code}", 

145 details={ 

146 "authoritative": True, 

147 "acceptance_test_ref": ref, 

148 "command": cmd, 

149 "output_tail": output, 

150 }, 

151 )
