1"""Apply persona-emitted context updates to an app repo. 

2 

3Phase-0 scope is intentionally minimal: a typed list of updates is applied 

4directly to the filesystem after validating every target path against the 

5canonical-paths whitelist + the forbidden-paths blacklist. Phase 2 will wire 

6this to actual persona output parsing. 

7""" 

8 

9from __future__ import annotations 

10 

11from dataclasses import dataclass 

12from pathlib import Path 

13from typing import Literal 

14 

15from factory.context.canonical_paths import is_canonical_doc_path, is_forbidden_doc_path 

16 

17Action = Literal["create", "rewrite"] 

18 

19 

20class ForbiddenContextPathError(ValueError): 

21 """Raised when a context update targets a non-canonical or forbidden path.""" 

22 

23 def __init__(self, path: str, reason: str) -> None: 

24 super().__init__(f"{path}: {reason}") 

25 self.path = path 

26 self.reason = reason 

27 

28 

29@dataclass(frozen=True) 

30class ContextUpdate: 

31 """A single context-file change. 

32 

33 Attributes: 

34 path: repo-relative path, e.g. ``context/modules/auth.md``. 

35 action: ``"create"`` (file must not exist) or ``"rewrite"`` (file may exist). 

36 content: full file content. Updater always writes the file whole — no 

37 line-level patching. Context is current-state-only; rewriting the whole 

38 file is the intended semantics. 

39 """ 

40 

41 path: str 

42 action: Action 

43 content: str 

44 

45 

46def apply_context_updates(updates: list[ContextUpdate], app_repo_path: Path) -> None: 

47 """Apply ``updates`` under ``app_repo_path``. 

48 

49 For every update, FIRST validate the path; raise 

50 ``ForbiddenContextPathError`` and apply NOTHING if any update fails 

51 validation (transactional in spirit — we want partial-write failures to be 

52 a separate, later-phase concern). 

53 

54 For ``action="create"``, the existing-file check is best-effort: a path that 

55 already exists raises ``FileExistsError``. 

56 """ 

57 repo = Path(app_repo_path) 

58 

59 # Validate all paths first; bail before writing if any fail. 

60 for upd in updates: 

61 if is_forbidden_doc_path(upd.path): 

62 raise ForbiddenContextPathError( 

63 upd.path, 

64 "path matches FORBIDDEN_DOC_PATTERNS (no ADRs, no changelogs, no history files)", 

65 ) 

66 if not is_canonical_doc_path(upd.path): 

67 raise ForbiddenContextPathError( 

68 upd.path, 

69 "path is not in CANONICAL_CONTEXT_PATHS (docs must live in the canonical layout)", 

70 ) 

71 

72 # Apply. 

73 for upd in updates: 

74 target = repo / upd.path 

75 target.parent.mkdir(parents=True, exist_ok=True) 

76 if upd.action == "create" and target.exists(): 

77 raise FileExistsError(f"create requested but path exists: {upd.path}") 

78 target.write_text(upd.content, encoding="utf-8")
