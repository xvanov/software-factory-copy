1"""Detector: retry_storm — surface repeated failures per (story, persona) pair. 

2 

3This module exposes ``retry_storm``, which groups failed run events by 

4``(story_id, persona)`` and returns per-group failure counts plus a 

5sample of error excerpts. The calling agent decides whether a given 

6failure count warrants action. 

7""" 

8 

9from __future__ import annotations 

10 

11import json 

12from datetime import datetime 

13from pathlib import Path 

14 

15 

16def retry_storm( 

17 *, 

18 root: Path, 

19 persona: str | None = None, 

20 story_id: int | None = None,

22) -> list[dict]: 

23 """Group failed run events by (story_id, persona) and return per-group counts. 

24 

25 Reads ``state/events/runs.ndjson`` and aggregates failure events since 

26 *since*, optionally filtered by *persona* and/or *story_id*. 

27 

28 Parameters 

29 ---------- 

30 root: 

31 Factory root directory. 

32 persona: 

33 If provided, restrict to events with this persona value. 

34 story_id: 

35 If provided, restrict to events with this story_id value. 

36 since: 

37 Lower bound (inclusive) on the event ``ts`` field. 

38 

39 Returns 

40 ------- 

41 list[dict] 

42 One dict per ``(story_id, persona)`` group, sorted by 

43 ``failure_count`` descending. Each dict contains: 

44 

45 * ``story_id`` — int or None 

46 * ``persona`` — str 

47 * ``failure_count`` — int 

48 * ``first_ts`` — ISO-8601 string of earliest failure in group 

49 * ``last_ts`` — ISO-8601 string of latest failure in group 

50 * ``error_excerpts`` — list of up to 5 truncated error strings 

51 (≤200 chars each) 

52 

53 Empty list when no matching failures exist or the file is missing. 

54 """ 

55 stream = root / "state" / "events" / "runs.ndjson" 

56 if not stream.exists(): 

57 return [] 

58 

59 since_iso = since.isoformat() 

60 # groups: (story_id, persona) -> {"ts_list", "errors"} 

61 groups: dict[tuple, dict] = {} 

62 

63 with stream.open(encoding="utf-8") as fh: 

64 for raw in fh: 

65 raw = raw.strip() 

66 if not raw: 

67 continue 

68 try: 

69 rec = json.loads(raw) 

70 except json.JSONDecodeError: 

71 continue 

72 # Only failures 

73 if rec.get("success") is True or rec.get("success") == 1: 

74 continue 

75 ts = rec.get("ts", "") 

76 if ts < since_iso: 

77 continue 

78 rec_persona = rec.get("persona", "") 

79 rec_story_id = rec.get("story_id") 

80 # Apply optional filters 

81 if persona is not None and rec_persona != persona: 

82 continue 

83 if story_id is not None and rec_story_id != story_id: 

84 continue 

85 

86 key = (rec_story_id, rec_persona) 

87 if key not in groups: 

88 groups[key] = {"ts_list": [], "errors": []} 

89 groups[key]["ts_list"].append(ts) 

90 err = rec.get("error") or "" 

91 if err: 

92 groups[key]["errors"].append(str(err)[:200]) 

93 

94 results: list[dict] = [] 

95 for (sid, pname), data in groups.items(): 

96 ts_list = sorted(data["ts_list"]) 

97 excerpts = data["errors"][:5] 

98 results.append( 

99 { 

100 "story_id": sid, 

101 "persona": pname, 

102 "failure_count": len(ts_list), 

103 "first_ts": ts_list[0] if ts_list else "", 

104 "last_ts": ts_list[-1] if ts_list else "", 

105 "error_excerpts": excerpts, 

106 } 

107 ) 

108 

109 results.sort(key=lambda r: r["failure_count"], reverse=True) 

110 return results
