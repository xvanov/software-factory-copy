1"""Phase-6 scheduled persona runner + chain integration. 

2 

3Public entry point: ``run_scheduled_persona(persona, app, 

4software_factory_root, *, dry_run=False) -> ScheduledRunRecord``. 

5 

6What the function does: 

7 

8 1. Loads the persona prompt from ``factory/personas/<persona>.md``. 

9 2. Composes a context prelude (current-state + module map). 

10 3. Invokes ``runner.text_run`` (ralph/bug_hunter/security) or 

11 ``runner.sandbox_run`` (ux_auditor — needs the browser tool). 

12 4. Parses the structured JSON output. 

13 5. For each finding/drift, calls 

14 ``factory.directions.creator.create_direction`` to file a fresh 

15 direction directory under ``apps/<app>/directions/``. 

16 6. Persists a ``ScheduledRunRecord`` row. 

17 

18Dry-run is truly dry: no LLM call, no GitHub call, no real subprocesses. 

19The fixture path returns a deterministic ``ScheduledRunOutput`` per 

20persona so the CLI's ``--dry-run`` flag can exercise the full chain 

21end-to-end. 

22"""

239def _slugify_title(title: str) -> str: 

240 import re 

241 

242 s = re.sub(r"[^A-Za-z0-9]+", "-", title.strip().lower()).strip("-") 

243 return (s[:40] or "ralph-finding").strip("-") or "ralph-finding"

246# Statuses that mean a direction is finished/abandoned — a new duplicate for 

247# the same issue is fine once the prior one is closed out. 

248_TERMINAL_DIRECTION_STATUSES = frozenset( 

249 {"done", "complete", "completed", "merged", "abandoned", "superseded", "cancelled"} 

250)

252 

253def _normalize_title(title: str) -> str: 

254 return " ".join(title.lower().split()) 

255 

256 

257# Stopgap semantic dedup (audit 2026-07-18, leak 3 of 4): scheduler personas 

258# invent a fresh title every run for the SAME underlying finding ("Add CSRF 

259# protections to cookie-auth API routes" vs "...API flows"), so the 

260# exact-title check in ``_has_open_duplicate_direction`` lets duplicates 

261# through (CSRF filed as 2 directions, SSRF as 2, OAuth as 4, observed 

262# live). Rather than touch the persona prompt or add a schema field, extract 

263# a small, precise set of stable keywords from the normalized title+body and 

264# treat a new finding as a duplicate of an EXISTING NON-TERMINAL direction 

265# for the same app + same ``source`` + same ``type`` if they share a 

266# keyword. Deliberately narrow — only named, specific terms trigger a 

267# match, never generic words — so unrelated findings (e.g. csrf vs a11y) 

268# never collapse into each other. 

269_DEDUP_KEYWORD_GROUPS: dict[str, tuple[str, ...]] = { 

270 "csrf": ("csrf",), 

271 "ssrf": ("ssrf",), 

272 "oauth": ("oauth",), 

273 "xss": ("xss", "cross-site scripting"), 

274 "authz": ( 

275 "authz", 

276 "authorization", 

277 "object-level authorization", 

278 "access control", 

279 "object level authorization", 

280 ), 

281 "rate-limit": ("rate limit", "rate-limit", "rate limiting", "throttl", "abuse control"), 

282 "webhook": ("webhook",), 

283 "replay": ("replay",), 

284 "redaction": ("redact", "secret leak", "secret exposure", "secret sprawl"), 

285 "a11y": ("accessib", "a11y"), 

286 "proof-schema": ("proof schema", "proof-schema"), 

287}

289 

290def _extract_dedup_keywords(text: str) -> set[str]: 

291 """Extract the small set of stable dedup keywords present in ``text``. 

292 

293 Case-insensitive substring match against ``_DEDUP_KEYWORD_GROUPS``. 

294 Returns the set of canonical keyword ids (e.g. ``{"csrf"}``) whose any 

295 trigger phrase appears in ``text``. Empty text or no matches -> empty 

296 set, which never dedups anything (precise-by-default). 

297 """ 

298 normalized = text.lower() 

299 return { 

300 keyword 

301 for keyword, triggers in _DEDUP_KEYWORD_GROUPS.items() 

302 if any(trigger in normalized for trigger in triggers) 

303 }

306def _has_open_duplicate_direction( 

307 app: str, 

308 title: str, 

309 software_factory_root: Path, 

310 *, 

311 source: str | None = None, 

312 type_tag: str | None = None, 

313 body: str = "", 

314) -> bool: 

315 """True if an existing non-terminal direction is a duplicate of this finding. 

316 

317 Two independent checks, either of which is sufficient: 

318 

319 1. Exact match: a non-terminal direction with the same normalized 

320 title exists (original guard, app-wide — unchanged behavior when 

321 ``source``/``type_tag`` are omitted). 

322 2. Semantic (stopgap): when ``source`` and ``type_tag`` are both 

323 given, a non-terminal direction for the same app + same 

324 ``source`` + same ``type`` whose title+body shares a dominant 

325 dedup keyword (see ``_DEDUP_KEYWORD_GROUPS``) with this finding's 

326 title+body. 

327 

328 Scans ``apps/<app>/directions/*/`` reading each direction.md 

329 title/body and state.yaml status/source directly (cheap, no full 

330 parse). Errors on any single directory are ignored so a malformed 

331 sibling never blocks filing. 

332 """ 

333 import frontmatter as _frontmatter 

334 import yaml as _yaml 

335 

336 target = _normalize_title(title) 

337 target_keywords = _extract_dedup_keywords(f"{title}\n{body}") 

338 directions_dir = Path(software_factory_root) / "apps" / app / "directions" 

339 if not directions_dir.is_dir(): 

340 return False 

341 for d in directions_dir.iterdir(): 

342 md = d / "direction.md" 

343 if not md.is_file(): 

344 continue 

345 try: 

346 post = _frontmatter.load(str(md)) 

347 metadata = post.metadata or {} 

348 existing_title = str(metadata.get("title") or "") 

349 

350 state_path = d / "state.yaml" 

351 status = "created" 

352 existing_source: Any = None 

353 if state_path.is_file(): 

354 state = _yaml.safe_load(state_path.read_text(encoding="utf-8")) or {} 

355 status = str(state.get("status", "created")) 

356 existing_source = state.get("source") 

357 if status in _TERMINAL_DIRECTION_STATUSES: 

358 continue 

359 

360 if _normalize_title(existing_title) == target: 

361 return True 

362 

363 if ( 

364 target_keywords 

365 and source is not None 

366 and type_tag is not None 

367 and existing_source == source 

368 and str(metadata.get("type") or "") == type_tag 

369 ): 

370 existing_body = str(getattr(post, "content", "") or "") 

371 existing_keywords = _extract_dedup_keywords(f"{existing_title}\n{existing_body}") 

372 if target_keywords & existing_keywords: 

373 return True 

374 except Exception: 

375 continue 

376 return False

379def _file_finding_as_direction( 

380 *, 

381 persona: str, 

382 app: str, 

383 finding: dict[str, Any], 

384 software_factory_root: Path, 

385 dry_run: bool = False, 

386) -> Direction | None: 

387 """Create a direction directory for one finding. Returns the parsed Direction. 

388 

389 Falls back gracefully if the finding lacks a ``suggested_direction`` 

390 block (returns ``None``). 

391 

392 When ``dry_run`` is True, the direction is written to a scratch 

393 directory under ``<software_factory_root>/state/dry_run_scratch/`` so 

394 the canonical ``apps/<app>/directions/`` tree is NOT polluted. The 

395 parser still reads it so the caller sees a real ``Direction`` and 

396 asserts the same shape it would on a real run. 

397 """ 

398 suggested = finding.get("suggested_direction") 

399 if not isinstance(suggested, dict): 

400 return None 

401 title = str(suggested.get("title") or "").strip() 

402 type_tag = str(suggested.get("type") or "bug").strip() 

403 why = str(suggested.get("why") or "").strip() 

404 acceptance_raw = suggested.get("acceptance") or [] 

405 acceptance = [str(a) for a in acceptance_raw if str(a).strip()] 

406 if not title or not why: 

407 return None 

408 

409 # Dedup guard. Scheduled personas re-run on a schedule and re-surface the 

410 # same finding until it's fixed; without this an unresolved issue spawns a 

411 # new near-identical direction every run (observed 2026-07-06: ~38 duplicate 

412 # "resolve conflicted navigation context" directions). If a non-terminal 

413 # direction with the same normalized title already exists for this app, 

414 # skip filing another. Also catches the same finding re-titled slightly 

415 # differently each run (e.g. "...API routes" vs "...API flows") via the 

416 # keyword-based semantic check — see ``_has_open_duplicate_direction``. 

417 # (Real runs only — dry-run writes to a scratch tree.) 

418 if not dry_run and _has_open_duplicate_direction( 

419 app, 

420 title, 

421 software_factory_root, 

422 source=f"scheduled-{persona}", 

423 type_tag=type_tag, 

424 body=why, 

425 ): 

426 return None 

427 target_root = software_factory_root 

428 if dry_run: 

429 # Route every write under state/dry_run_scratch/ so apps/<app>/ 

430 # directions/ remains untouched on dry-run paths. 

431 scratch = Path(software_factory_root) / "state" / "dry_run_scratch" 

432 scratch.mkdir(parents=True, exist_ok=True) 

433 target_root = scratch 

434 # next_direction_id scans target_root/apps/<app>/directions/, so 

435 # ensure it exists. 

436 (scratch / "apps" / app / "directions").mkdir(parents=True, exist_ok=True) 

437 # UX auditor findings carry flow + step data; surface a flow.md so the 

438 # next slice (citation/parsing) has concrete filenames and step numbers 

439 # to reference. All other personas stay as-is — their findings are not 

440 # UI-specific. 

441 has_ui = persona == "ux_auditor" 

442 flow_steps: list[str] | None = None 

443 if has_ui: 

444 flow_name = str(finding.get("flow") or "").strip() 

445 step = finding.get("step") 

446 evidence = str(finding.get("evidence") or "").strip() 

447 suggestion = str(finding.get("suggestion") or "").strip() 

448 steps: list[str] = [] 

449 if flow_name: 

450 steps.append(f"Flow: {flow_name}") 

451 if isinstance(step, int): 

452 steps.append(f"Step: {step}") 

453 if evidence: 

454 steps.append(f"Evidence: {evidence}") 

455 if suggestion: 

456 steps.append(f"Suggestion: {suggestion}") 

457 if steps: 

458 flow_steps = steps 

459 

460 created = create_direction( 

461 app, 

462 title=title, 

463 type_tag=type_tag, 

464 why=why, 

465 has_ui=has_ui, 

466 flow_steps=flow_steps, 

467 has_api=False, 

468 api_spec_lines=None, 

469 acceptance=acceptance, 

470 # Scheduled personas (bug_hunter/ralph/security) file findings the 

471 # factory itself should investigate and fix. They have no 

472 # user_flow/api_spec — a bug report isn't a feature spec — so with 

473 # explore=False they ALWAYS failed the backpressure gate and produced 

474 # zero stories (observed 2026-07-06: every scheduled-* direction stuck 

475 # at needs-direction, nothing ever built). explore=True is the correct 

476 # channel: "here's a problem, investigate and fix it", which the PM can 

477 # decompose without a spec. This is what makes idle bug-hunting actually 

478 # ship fixes instead of accumulating dead directions. 

479 explore=True, 

480 attach_files=None, 

481 software_factory_root=target_root, 

482 source=f"scheduled-{persona}{'-dry' if dry_run else ''}", 

483 ) 

484 return created.direction

492def run_scheduled_persona( 

493 persona: str, 

494 app: str, 

495 software_factory_root: Path, 

496 *, 

497 dry_run: bool = False, 

498 db_path: Path | None = None, 

499 fixture_output: dict[str, Any] | None = None, 

500) -> ScheduledRunOutput: 

501 """Execute one scheduled persona run. 

502 

503 On ``dry_run=True`` no LLM is called, no GH issue is opened, and the 

504 canonical ``apps/<app>/directions/`` tree is NOT mutated — directions 

505 are written to ``<root>/state/dry_run_scratch/apps/<app>/directions/`` 

506 so the CLI's --dry-run is end-to-end testable without API keys or 

507 pollution. The ``cron_schedules.last_run`` column is also untouched 

508 on a dry-run; only real runs update it. Pass ``fixture_output`` to 

509 override the default per-persona fixture. 

510 

511 On real-run, ``text_run`` is invoked with the persona prompt + the 

512 composed context prelude; the output is parsed as JSON and findings 

513 are filed. 

514 

515 Rate-limit gate runs FIRST: ``can_dispatch(persona, app, state, 

516 settings)`` consults the per-persona daily-run cap recorded in 

517 ``factory_settings.yaml`` (``rate_limits.<persona>_runs_per_day``). 

518 When refused, the run records a ``rejected`` row with the canonical 

519 ``rejected_reason`` and returns immediately (no LLM, no directions). 

520 """ 

521 root = Path(software_factory_root) 

522 db = db_path or (root / "state" / "factory.db") 

523 started = datetime.now(UTC) 

524 findings_key = _PERSONA_FINDINGS_KEY.get(persona) 

525 if findings_key is None: 

526 return _record_and_return( 

527 persona=persona, 

528 app=app, 

529 duration_s=0.0, 

530 findings_count=0, 

531 directions_filed=[], 

532 status="errored", 

533 error=f"unknown_scheduled_persona: {persona!r}", 

534 dry_run=dry_run, 

535 raw_output={}, 

536 db_path=db, 

537 ) 

538 

539 # Rate-limit gate. Phase 6 personas (ralph/bug_hunter/security/ 

540 # ux_auditor) each have a per-day cap in 

541 # ``factory_settings.yaml.rate_limits``; ``can_dispatch`` returns 

542 # ``rejected_reason="<persona>_rate_limit_exceeded"`` when the cap is 

543 # hit. Runs that are refused are recorded with status="rejected" so 

544 # the audit trail captures the attempt. 

545 from factory.settings.enforcer import can_dispatch 

546 from factory.settings.loader import load_settings 

547 from factory.settings.modes import get_mode 

548 from factory.settings.spend import ( 

549 hour_spend_usd, 

550 persona_runs_today, 

551 today_spend_usd, 

552 ) 

553 

554 if persona in _PERSONA_FINDINGS_KEY: 

555 settings = load_settings(root) 

556 state = { 

557 "mode": get_mode(root, db_path=db), 

558 "global_in_flight": 0, 

559 "app_in_flight": 0, 

560 "today_spend_usd": today_spend_usd(root, db_path=db), 

561 "hour_spend_usd": hour_spend_usd(root, db_path=db), 

562 "open_prs_for_app": None, 

563 "failing_ci_count": None, 

564 "pm_invocations_last_hour": 0, 

565 f"{persona}_runs_today": persona_runs_today(persona, root, db_path=db), 

566 } 

567 decision = can_dispatch(persona, app, state, settings) 

568 if not decision.allowed: 

569 duration = (datetime.now(UTC) - started).total_seconds() 

570 return _record_and_return( 

571 persona=persona, 

572 app=app, 

573 duration_s=duration, 

574 findings_count=0, 

575 directions_filed=[], 

576 status="rejected", 

577 error=decision.rejected_reason, 

578 dry_run=dry_run, 

579 raw_output={}, 

580 db_path=db, 

581 ) 

582 

583 raw: dict[str, Any] 

584 error: str | None = None 

585 if dry_run: 

586 raw = fixture_output if fixture_output is not None else _dry_run_fixture(persona, app) 

587 else: 

588 try: 

589 raw = _live_run(persona, app, root) 

590 except Exception as exc: # noqa: BLE001 - capture all exceptions for audit 

591 duration = (datetime.now(UTC) - started).total_seconds() 

592 return _record_and_return( 

593 persona=persona, 

594 app=app, 

595 duration_s=duration, 

596 findings_count=0, 

597 directions_filed=[], 

598 status="errored", 

599 error=str(exc), 

600 dry_run=False, 

601 raw_output={}, 

602 db_path=db, 

603 ) 

604 

605 findings = raw.get(findings_key) or [] 

606 if not isinstance(findings, list): 

607 findings = [] 

608 

609 directions_filed: list[str] = [] 

610 for finding in findings: 

611 if not isinstance(finding, dict): 

612 continue 

613 try: 

614 direction = _file_finding_as_direction( 

615 persona=persona, 

616 app=app, 

617 finding=finding, 

618 software_factory_root=root, 

619 dry_run=dry_run, 

620 ) 

621 except Exception as exc: # noqa: BLE001 - one bad finding shouldn't kill the run 

622 error = f"direction_create_failed: {exc}" 

623 continue 

624 if direction is not None: 

625 directions_filed.append(direction.id) 

626 

627 duration = (datetime.now(UTC) - started).total_seconds() 

628 return _record_and_return( 

629 persona=persona, 

630 app=app, 

631 duration_s=duration, 

632 findings_count=len(findings), 

633 directions_filed=directions_filed, 

634 status="dry_run" if dry_run else "ok",

637 raw_output=raw, 

638 db_path=db,

642def _record_and_return( 

643 *, 

644 persona: str, 

645 app: str, 

646 duration_s: float, 

647 findings_count: int, 

648 directions_filed: list[str], 

649 status: str, 

650 error: str | None, 

651 dry_run: bool, 

652 raw_output: dict[str, Any], 

653 db_path: Path, 

654) -> ScheduledRunOutput: 

655 eng = _engine(db_path) 

656 rec = ScheduledRunRecord( 

657 persona=persona, 

658 app=app, 

659 duration_s=round(duration_s, 4), 

660 findings_count=findings_count, 

661 directions_filed_json=json.dumps(directions_filed), 

662 status=status, 

663 error=error, 

664 dry_run=dry_run, 

665 ) 

666 with Session(eng) as session: 

667 session.add(rec) 

668 session.commit() 

669 session.refresh(rec) 

670 # Update the schedule's last-run timestamp so the cron scheduler 

671 # doesn't re-fire this slot. Best-effort; failure here is recorded 

672 # but doesn't fail the run. 

673 # 

674 # Dry-run is truly dry: NEVER mutate cron_schedules.last_run on a 

675 # dry-run path. Otherwise an operator probing with --dry-run could 

676 # cause the real cron loop to think a slot already fired and skip it. 

677 if not dry_run: 

678 try: 

679 from factory.scheduler.cron import load_schedules, upsert_schedule_row 

680 

681 for sched in load_schedules(Path(db_path).parent.parent): 

682 if sched.persona == persona: 

683 upsert_schedule_row( 

684 name=sched.name, 

685 cron_expr=sched.cron_expr, 

686 last_run=datetime.now(UTC).isoformat(), 

687 last_status=status, 

688 db_path=db_path, 

689 ) 

690 except Exception: # noqa: BLE001 - schedule update is best-effort 

691 pass 

692 return ScheduledRunOutput( 

693 persona=persona, 

694 app=app, 

695 findings_count=findings_count, 

696 directions_filed=directions_filed, 

697 duration_s=round(duration_s, 4), 

698 status=status, 

699 error=error, 

700 dry_run=dry_run, 

701 raw_output=raw_output, 

702 )

705def _collect_flow_artifacts(app: str, software_factory_root: Path) -> list[tuple[str, str]]: 

706 """Collect flow.md files from all direction directories for ``app``. 

707 

708 Returns a list of (label, content) tuples. The label is 

709 ``<direction-id-slug>/flow.md`` so the UX auditor can cite it by name. 

710 Empty directories and missing files are silently skipped. 

711 """ 

712 directions_dir = Path(software_factory_root) / "apps" / app / "directions" 

713 if not directions_dir.is_dir(): 

714 return [] 

715 artifacts: list[tuple[str, str]] = [] 

716 for d in sorted(directions_dir.iterdir()): 

717 if not d.is_dir(): 

718 continue 

719 flow_md = d / "flow.md" 

720 if not flow_md.is_file(): 

721 continue 

722 try: 

723 content = flow_md.read_text(encoding="utf-8").strip() 

724 if content: 

725 artifacts.append((f"{d.name}/flow.md", content)) 

726 except Exception: # noqa: BLE001 - best-effort collection 

727 continue 

728 return artifacts

731def _extract_http_urls(text: str) -> list[str]: 

732 """Return distinct HTTP(S) URLs found in arbitrary shell text.""" 

733 import re 

734 

735 matches = re.findall(r"https?://[^\s`\"')]+", text) 

736 return sorted({m.rstrip(".,;:") for m in matches})

739def _build_ux_auditor_context(app: str, software_factory_root: Path) -> str: 

740 """Build UX-auditor runtime input context for scheduled dispatch. 

741 

742 The scheduled UX run MUST carry at least one concrete ``flow.md`` artifact; 

743 if none are available, the scheduler refuses dispatch instead of sending a 

744 guess-based prompt. 

745 """ 

746 from datetime import UTC, datetime 

747 

748 from factory.app_config import load_app_config 

749 

750 parts: list[str] = ["## Scheduled UX Audit Runtime Inputs\n"] 

751 

752 # -- Flow artifacts (AC1.1) -- 

753 flow_artifacts = _collect_flow_artifacts(app, software_factory_root) 

754 if not flow_artifacts: 

755 raise ValueError( 

756 f"scheduled ux_auditor input for app '{app}' requires at least one flow.md artifact" 

757 ) 

758 

759 parts.append("\n### Flow Artifacts\n") 

760 parts.append( 

761 f"_Collected {len(flow_artifacts)} flow.md file(s) from prior directions. " 

762 "Use these concrete filenames and numbered steps during the audit._\n" 

763 ) 

764 for label, content in flow_artifacts: 

765 parts.append(f"\n#### {label}\n") 

766 parts.append(content + "\n") 

767 

768 # -- App URL context (AC1.2) -- 

769 parts.append("\n### App URL Context\n") 

770 try: 

771 cfg = load_app_config(app, software_factory_root) 

772 parts.append(f"- App name: `{cfg.name}`\n") 

773 parts.append(f"- Repo: `{cfg.repo}`\n") 

774 

775 url_candidates: list[str] = [] 

776 if cfg.deploy.health_check_command: 

777 url_candidates.extend(_extract_http_urls(cfg.deploy.health_check_command)) 

778 if cfg.deploy.smoke_test_command: 

779 url_candidates.extend(_extract_http_urls(cfg.deploy.smoke_test_command)) 

780 

781 if cfg.deploy.enabled: 

782 parts.append("- Deploy: **enabled**\n") 

783 if cfg.deploy.health_check_command: 

784 parts.append(f" - Health check command: `{cfg.deploy.health_check_command}`\n") 

785 if url_candidates: 

786 for url in sorted(set(url_candidates)): 

787 parts.append(f" - App URL candidate: `{url}`\n") 

788 else: 

789 parts.append(" - App URL candidate: not declared in deploy commands\n") 

790 else: 

791 parts.append("- Deploy: **disabled** (no live URL configured)\n") 

792 except Exception: # noqa: BLE001 - best-effort 

793 parts.append("- (could not load app config)\n") 

794 

795 # -- Runtime context (AC1.3) -- 

796 parts.append("\n### Runtime Context\n") 

797 parts.append(f"- Timestamp (UTC): `{datetime.now(UTC).isoformat()}`\n") 

798 parts.append(f"- Software factory root: `{software_factory_root}`\n") 

799 parts.append(f"- Target app: `{app}`\n") 

800 parts.append("- Scheduler transport: `text_run`\n") 

801 

802 return "".join(parts)

805def _live_run(persona: str, app: str, software_factory_root: Path) -> dict[str, Any]: 

806 """Compose context + persona prompt + dispatch via runner. 

807 

808 Ralph/bug_hunter/security/ux_auditor all use ``text_run`` for v1; the 

809 sandbox path (browser tool) is reserved for a future ux_auditor 

810 enhancement when the live deploy URL exists. 

811 """ 

812 from factory.app_config import load_app_config, resolve_app_repo_path 

813 from factory.context.loader import compose_context_prelude 

814 from factory.runner import text_run 

815 

816 cfg = load_app_config(app, software_factory_root) 

817 prelude = compose_context_prelude( 

818 persona, 

819 app_repo_path=resolve_app_repo_path(cfg, software_factory_root), 

820 task_scope=None, 

821 ) 

822 persona_md_path = Path(__file__).resolve().parent.parent / "personas" / f"{persona}.md" 

823 persona_prompt = persona_md_path.read_text(encoding="utf-8") if persona_md_path.exists() else "" 

824 

825 # UX auditor: enrich the prompt with flow artifacts, app URL, and runtime context. 

826 if persona == "ux_auditor": 

827 ux_context = _build_ux_auditor_context(app, software_factory_root) 

828 prompt = f"{persona_prompt}\n\n{ux_context}\n\n# Context prelude\n\n{prelude}\n" 

829 else: 

830 prompt = f"{persona_prompt}\n\n# Context prelude\n\n{prelude}\n" 

831 

832 model = route(persona) 

833 max_tokens = _OUTPUT_TOKEN_CAP.get(persona, 2048) 

834 result = text_run( 

835 persona, 

836 prompt, 

837 model, 

838 schema={"type": "object"}, 

839 max_tokens=max_tokens, 

840 # These app-level personas (ralph/bug_hunter/security/ux_auditor) run 

841 # ahead of any story, so story_id/direction_id are legitimately 

842 # unknown here — but ``app`` IS known and must be stamped so 

843 # ``factory audit`` can roll spend up per app instead of leaving it 

844 # in the unattributed bucket. 

845 app=app, 

846 ) 

847 if isinstance(result, dict): 

848 return result 

849 # text_run returned a raw string (schema not respected); try to parse. 

850 try: 

851 parsed = json.loads(result) 

852 if isinstance(parsed, dict): 

853 return parsed 

854 except Exception: 

855 pass 

856 return {}
