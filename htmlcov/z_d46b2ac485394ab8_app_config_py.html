64class AppGatesConfig(BaseModel): 

65 """Per-app gate commands consumed by the auto-merge worker (Phase 4). 

66 

67 Every field is optional: a missing command means "skip this gate". The 

68 factory itself is stack-agnostic — these strings are executed verbatim 

69 by the gate handler when the worker is in real-run mode, and only flag 

70 lookups are done in dry-run. 

71 """ 

72 

73 lint_command: str | None = None 

74 format_check_command: str | None = None 

75 type_check_command: str | None = None 

76 test_command: str | None = None 

77 coverage_command: str | None = None 

78 e2e_command: str | None = None 

79 mutation_testing: bool = False 

80 # Whether a WORKING end-to-end/browser harness actually exists in the app 

81 # (Playwright installed + config + the stack runnable in the sandbox). A 

82 # configured ``e2e_command`` does NOT imply this — sacrifice declares 

83 # ``npx playwright test`` but has no runnable harness, so test_designer 

84 # mandating Playwright produced harness-breakage "reds" that deadlocked 

85 # every frontend/E2E story. When False, the test_designer must NOT require 

86 # E2E/Playwright and should scope to the backend test_command instead. 

87 e2e_harness_ready: bool = False 

88 

89 # Runtime smoke gate (Karpathy Layer-2 "external signal", D002). A command 

90 # that BOOTS the running product and exercises one real user journey 

91 # (e.g. docker compose up + a scripted sign-up → login → core-action pass). 

92 # Distinct from ``test_command`` (unit/integration, app never starts) and 

93 # from ``e2e_command`` (declared but historically unrunnable). The factory 

94 # shipped a full backlog green while the app could not log in precisely 

95 # because nothing booted it; this gate is the oracle that closes that class. 

96 smoke_command: str | None = None 

97 # Whether a WORKING smoke harness actually exists (stack runnable in the 

98 # sandbox + the scripted journey passes). Only when True does ``smoke-green`` 

99 # become a merge-REQUIRED gate for this app — keeping the rollout per-app 

100 # opt-in so apps without a harness are unaffected (no new merge blocks, 

101 # avoiding the PRs 110/111 "every merge blocked" regression). 

102 smoke_harness_ready: bool = False 

103 

104 # WS1.2 independent acceptance oracle. When True, the chain authors an 

105 # acceptance test from each story's direction acceptance criteria (the SPEC 

106 # ONLY, blind to the dev's code/tests) at spawn time, stores it in factory 

107 # state OUTSIDE the dev worktree, and the ``acceptance-verified`` gate copies 

108 # it into the merge-candidate checkout and runs it as a REQUIRED gate. Off by 

109 # default so the rollout is per-app opt-in — an app that hasn't enabled it 

110 # sees no new merge blocks (mirrors the ``smoke_harness_ready`` rollout). The 

111 # gate is required only for stories that actually got an oracle authored 

112 # (ACs present + this flag on); legacy / no-AC stories are never blocked. 

113 acceptance_oracle: bool = False 

114 # Command template the acceptance gate runs, with ``{test_file}`` substituted 

115 # for the copied-in test's path (relative to the checkout root). Defaults to 

116 # ``python -m pytest {test_file} -q`` when unset; apps whose suite needs a 

117 # wrapper (e.g. ``uv run pytest {test_file} -q``) override it here so the 

118 # oracle runs against the app's real python env. 

119 acceptance_test_command: str | None = None 

120 

121 # Flaky-test quarantine (WS4.4). When True, the ``tests-green`` real-run gate 

122 # routes a RED ``test_command`` through flake detection: any failing test is 

123 # re-run in isolation; a test that flaps (fails then passes) is QUARANTINED 

124 # (recorded to ``state/flake_quarantine.json`` + surfaced as an event/direction) 

125 # and no longer blocks the merge, while a test that fails CONSISTENTLY is a 

126 # real regression and still blocks. Off by default so the rollout is per-app 

127 # opt-in and never widens what passes the gate without an explicit choice — a 

128 # bug in flake classification could otherwise let a real red through, the 

129 # exact false-green this whole tier fights. 

130 flake_quarantine: bool = False 

131 # How many times a failing test is re-run in isolation before it is declared a 

132 # consistent (real) failure. A test that passes on ANY of these reruns flapped 

133 # and is quarantined. This is NOT retry-until-green (which manufactures 

134 # false-greens): a pass only ever DOWNGRADES a red to quarantined-non-blocking, 

135 # never upgrades it to a clean pass, and the flake stays surfaced as debt. 

136 flake_rerun_count: int = 3

139class AppConfig(BaseModel): 

140 name: str 

141 repo: str # "owner/name" 

142 default_branch: str = "main" 

143 context_dir: str = "context" 

144 # Path to the actual app source tree, relative to the factory root. 

145 # Default ``../<name>`` matches the convention "factory at 

146 # ``~/software-factory/``, apps at ``~/<name>/`` (siblings)". Personas 

147 # read context from this path, NOT from ``apps/<name>/`` inside the 

148 # factory (which only holds the per-app config + directions + state). 

149 app_repo_path: str = "" 

150 deploy: DeployConfig = Field(default_factory=DeployConfig) 

151 gates: AppGatesConfig = Field(default_factory=AppGatesConfig) 

152 models: dict[str, str] = Field(default_factory=dict) # persona overrides 

153 

154 # Self-tick guard (Tier 3 — FACTORY-SELF-TICK). When this app builds the 

155 # factory's OWN repo (``apps/factory``), pm-sync will NOT turn its pending 

156 # directions into chain stories unless this flag is True. OFF by default so 

157 # merely bootstrapping ``apps/factory`` (config + directions on disk) never 

158 # silently starts the factory ticking on itself — the orchestrator enables 

159 # self-tick deliberately by flipping this to True. It is inert for every 

160 # non-factory app (their merges are not self-edits; see 

161 # ``auto_merge._story_targets_factory_repo``). The chain-side staging gate 

162 # is independent of this flag: even when self-tick is enabled, a self-edit 

163 # still must pass staging before it can touch the live factory. 

164 self_tick_enabled: bool = False

175def load_app_config(app: str, software_factory_root: Path) -> AppConfig: 

176 """Load and validate ``apps/<app>/config.yaml`` from the factory root. 

177 

178 If ``app_repo_path`` is unset in the YAML, it defaults to ``../<name>`` 

179 relative to the factory root (e.g. factory at ``~/software-factory/`` 

180 and apps as sibling directories at ``~/<name>/``). 

181 """ 

182 cfg_path = Path(software_factory_root) / "apps" / app / "config.yaml" 

183 if not cfg_path.exists(): 

184 raise FileNotFoundError(f"App config missing: {cfg_path}. Expected apps/<app>/config.yaml.") 

185 raw = yaml.safe_load(cfg_path.read_text(encoding="utf-8")) or {} 

186 if not isinstance(raw, dict): 

187 raise ValueError(f"{cfg_path}: top-level must be a YAML mapping") 

188 if not raw.get("app_repo_path"): 

189 # Mirror the documented convention: app source lives at a sibling 

190 # of the factory root, named by the app. 

191 raw["app_repo_path"] = f"../{raw.get('name') or app}" 

192 return AppConfig.model_validate(raw)

195def resolve_app_repo_path(cfg: AppConfig, software_factory_root: Path) -> Path: 

196 """Resolve ``cfg.app_repo_path`` against the factory root. 

197 

198 Absolute paths are returned unchanged; relative paths are anchored at 

199 ``software_factory_root``. The result is NOT required to exist — 

200 callers handle the "no app tree yet" case (e.g. context loader emits 

201 the NO CONTEXT AVAILABLE notice). 

202 """ 

203 raw = (cfg.app_repo_path or "").strip() 

204 if not raw: 

205 raw = f"../{cfg.name}" 

206 p = Path(raw) 

207 if p.is_absolute(): 

208 return p 

209 return (Path(software_factory_root) / p).resolve()

212def list_apps(software_factory_root: Path) -> list[dict[str, object]]: 

213 """Discover every ``apps/*/config.yaml`` and return one summary dict per app. 

214 

215 Each dict contains at least: ``name``, ``repo``, ``self_tick_enabled``, 

216 ``deploy_enabled`` — reading effective values from the app's config.yaml. 

217 The function is pure: it never mutates config, filesystem, or runtime state. 

218 """ 

219 apps_dir = Path(software_factory_root) / "apps" 

220 if not apps_dir.is_dir(): 

221 return [] 

222 

223 result: list[dict[str, object]] = [] 

224 for cfg_path in sorted(apps_dir.glob("*/config.yaml")): 

225 app_name = cfg_path.parent.name 

226 cfg = load_app_config(app_name, software_factory_root) 

227 result.append( 

228 { 

229 "name": cfg.name, 

230 "repo": cfg.repo, 

231 "self_tick_enabled": cfg.self_tick_enabled, 

232 "deploy_enabled": cfg.deploy.enabled, 

233 } 

234 ) 

235 return result
