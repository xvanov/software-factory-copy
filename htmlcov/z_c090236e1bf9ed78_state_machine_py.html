1"""TDD chain state machine. 

2 

3The chain advances each ``StoryRecord`` through a fixed sequence of states. 

4Each transition is driven by an *event* (handler completion, webhook 

5arrival, retry timer) carrying an optional payload (handler result, GitHub 

6event body, error string). 

7 

8This module is pure: ``advance(story, event, payload) -> StoryState`` only 

9computes the next state from the current state and the event. Handlers in 

10``factory/chain/handlers.py`` are responsible for the side effects 

11(LLM calls, GitHub API calls, file writes); they consume the next-state 

12from ``advance`` to know what to persist. 

13 

14State names mirror the plan section "TDD chain (the heart of the 

15workflow)". The story-record fields capture everything a handler needs to 

16resume from any state (test_plan_json, dev_retries, current_model_tier, 

17etc). 

18"""

29class StoryState(StrEnum): 

30 """Every state a story can be in. 

31 

32 Two chain variants share the same enum; the orchestrator picks the 

33 starting handler based on ``story.chain_kind``: 

34 

35 * ``tdd`` (default): SM → dev → reviewer → tech_writer → docs_enforcer → 

36 PR_OPEN. The dev persona writes the production code AND its tests in one 

37 pass (Loop-4, dev-owns-tests); there is no separate test-design / 

38 test-implementation phase. 

39 * ``docs``: docs_sm → docs_onboarder → docs_enforcer → PR_OPEN. Used for 

40 stories whose deliverable is canonical documentation (e.g. the initial 

41 ``context/`` bootstrap). Skips the code+test loop entirely. 

42 

43 Both variants converge at ``DOCS_ENFORCER_CHECK`` and onward; the 

44 enforcer + PR + deploy states are shared. 

45 """

73 # Hard convergence guard: when the dev<->reviewer loop fails to converge 

74 # within _MAX_REVIEW_CYCLES reviewer passes, the reviewer handler routes 

75 # here instead of bouncing back to REVIEWER_REQUESTED_CHANGES. Terminal 

76 # (no outgoing transition) so the orchestrator stops dispatching the 

77 # story and it surfaces for human review instead of looping indefinitely. 

78 BLOCKED_REVIEW_NONCONVERGENT = "blocked_review_nonconvergent" 

79 # WS1.1 global per-story budget circuit breaker. The composed loops (dev 

80 # retries, reviewer cycles, auto-recovery re-dispatch, CI-fix) each have 

81 # their OWN counter but no aggregate per-story ceiling, so a pathological 

82 # story can burn dev(6)*review(6)+recovery(2)+CI-fix(3) loops of spend. 

83 # When a story's ``total_attempts`` or ``total_spend_usd`` crosses the 

84 # per-story cap, the orchestrator routes it here. Terminal (no outgoing 

85 # transition) so a broken story stops burning spend and surfaces for a 

86 # human with the evidence event the breaker emits. 

87 # 

88 # OPERATOR RESET (this is a terminal sink and its worktree gets pruned): 

89 # to re-run a story parked here, move it back to a live dispatch state 

90 # (e.g. ``sm_done`` to replay dev→review→merge, or ``story_created`` from 

91 # scratch) AND zero the accumulators (``total_attempts = 0``, 

92 # ``total_spend_usd = 0.0``) — otherwise the pre-dispatch breaker re-trips 

93 # immediately. To raise the ceiling globally instead, bump 

94 # ``caps.per_story_attempts`` / ``caps.per_story_spend_usd`` in 

95 # factory_settings.yaml. There is deliberately no auto-recovery path 

96 # (this state is absent from ``_AUTO_RECOVERABLE_STATES``). 

97 BLOCKED_BUDGET_EXCEEDED = "blocked_budget_exceeded" 

98 # Phase 7 dual-draft loser sink. An ambiguous/(explore) direction spawns 

99 # TWO ``draft-alternative`` siblings (slugs ``…-alt-a`` / ``…-alt-b``); only 

100 # ONE should ship. When the winner's PR merges, the loser is parked here by 

101 # ``dual_draft.close_abandoned_draft_sibling`` so the chain stops driving it 

102 # (otherwise a loser still in dev/review would open its own PR and merge a 

103 # redundant second interpretation to main — observed: direction 007 shipped 

104 # BOTH PR #67 and #69). TERMINAL: no outgoing transition, so ``is_terminal`` 

105 # reports True, ``_dispatch_for_story`` returns None, and it is absent from 

106 # ``auto_merge._MERGEABLE_STATES``. A supersede can fire from ANY in-flight 

107 # state, so the caller sets this via a DIRECT state assignment rather than a 

108 # state-machine ``advance()`` edge — deliberately NOT wired as an EVENT_* 

109 # transition from every source state. 

110 SUPERSEDED_BY_SIBLING = "superseded_by_sibling"

114 """Per-story chain record, persisted in ``state/factory.db.stories``. 

115 

116 One row per child_story spawned from a Direction's PM result. Carries 

117 every handler input/output and enough audit data to resume the chain 

118 after a crash. 

119 """

127 slug: str 

128 scope: str # frontend | backend | infra | test | docs 

129 state: str = Field(default=StoryState.STORY_CREATED.value, index=True) 

130 # Which chain variant drives this story. ``tdd`` is the historical 

131 # default; ``docs`` routes through docs_sm → docs_onboarder → 

132 # docs_enforcer for documentation-only deliverables. The orchestrator 

133 # reads this when dispatching out of STORY_CREATED.

135 github_issue_number: int | None = None 

136 github_branch: str | None = None 

137 github_pr_number: int | None = None

139 sm_result_json: str | None = None # JSON-serialized SM persona output 

140 # Retained: still rendered into the REVIEWER's prompt (handlers.handle_review) 

141 # as the "## Test plan" section, even though the test_designer persona that 

142 # once populated it was removed in the Loop-4 collapse (usually None now). 

143 test_plan_json: str | None = None 

144 reviewer_result_json: str | None = None 

145 # JSON-serialised list of prior review cycles (capped to the last 4): 

146 # ``[{cycle, verdict, score, findings: [{severity, location, what, 

147 # regression}], test_quality_findings: [...]}]``. Rendered into the 

148 # REVIEWER's re-review prompt (making the finality rule enforceable — the 

149 # reviewer otherwise has no memory of what it previously said) and into 

150 # DEV's prompt as an "already addressed, do not regress" digest. 

151 reviewer_history_json: str | None = None 

152 tech_writer_result_json: str | None = None 

153 dev_retries: int = 0 

154 # JSON-serialised list of prior dev attempts on this story. Each entry: 

155 # ``{"attempt": N, "ts": "...", "test_output_tail": "...", 

156 # "files_touched": [...], "summary": "..."}``. Carried forward into 

157 # the next dev sandbox's initial message so the LLM sees what it tried 

158 # and what failed instead of re-discovering dead ends from scratch. 

159 dev_attempts_json: str | None = None 

160 # WS4.2 resume-from-checkpoint marker. Set to a small JSON blob 

161 # (``{"outcome": "green", "attempt": N, "ts": ...}``) by the dev handler the 

162 # instant a sandbox returns GREEN — persisted BEFORE the state advances out 

163 # of ``dev_in_progress`` and cleared the instant the advance is persisted. 

164 # It therefore survives ONLY a tick that died in the tiny window between 

165 # "sandbox finished green" and "state advanced". On the next dev dispatch the 

166 # handler reads this marker and resumes to ``tests_green`` WITHOUT re-running 

167 # the (already-complete, already-in-the-worktree) dev LLM. None in steady 

168 # state. See ``handlers._try_resume_dev_from_checkpoint``. 

169 dev_step_checkpoint: str | None = None 

170 # Hard convergence guard counter: incremented each time the reviewer 

171 # returns a request-changes verdict in handle_review. When it reaches 

172 # ``_MAX_REVIEW_CYCLES`` the story is routed to 

173 # ``BLOCKED_REVIEW_NONCONVERGENT`` instead of looping back to dev, so a 

174 # non-converging dev<->reviewer ping-pong cannot burn budget unbounded. 

175 reviewer_cycles: int = 0 

176 # WS1.1 global per-story budget circuit breaker accumulators. Unlike the 

177 # per-loop counters above (dev_retries, reviewer_cycles) these span EVERY 

178 # loop the story passes through — dev retries, reviewer cycles, tech_writer, 

179 # docs, auto-recovery re-dispatch, CI-fix — giving one aggregate ceiling no 

180 # single loop's counter can see. ``total_attempts`` is bumped once per 

181 # handler dispatch by the orchestrator; ``total_spend_usd`` is refreshed 

182 # from the D003 per-run ledger (runs.cost_usd attributed to this story) 

183 # after each dispatch. When either crosses the per-story cap the 

184 # orchestrator advances the story to BLOCKED_BUDGET_EXCEEDED. 

185 total_attempts: int = 0 

186 total_spend_usd: float = 0.0 

187 # WS1.1 advance-decay high-water mark: the highest happy-path progress 

188 # ordinal (see orchestrator._STATE_PROGRESS_ORDINAL) this story has ever 

189 # reached. When a dispatch advances the story to a state whose ordinal 

190 # EXCEEDS this mark — genuine, monotonic forward progress, not a dev<->review 

191 # oscillation that re-treads states already seen — the orchestrator resets 

192 # ``total_attempts`` (the spend cap stays the hard ceiling). This keeps a 

193 # story that is still making progress from tripping the attempt breaker on a 

194 # poisoned historical count, while a stuck/oscillating story still exhausts 

195 # it. Default 0 so a story's first real state always registers. 

196 max_progress_ordinal: int = 0 

197 current_model_tier: str = "standard" # standard | hard 

198 created_at: str = Field(default_factory=lambda: datetime.now(UTC).isoformat()) 

199 updated_at: str = Field(default_factory=lambda: datetime.now(UTC).isoformat()) 

200 error: str | None = None 

201 # Phase 3: last cap/mode rejection reason emitted by the dispatcher. 

202 last_rejection_reason: str | None = None 

203 # D002 Karpathy Layer-2 runtime verifier. Set True after the dev's sandbox 

204 # boots the product and the scripted smoke journey passes. Read by the 

205 # ``smoke-green`` gate in dry-run; None means "not run yet". 

206 smoke_passed: bool | None = None 

207 # Item 4 — harness precheck. Set True after a one-shot pytest 

208 # collect+exit pass succeeds against the per-story worktree (with 

209 # ONLY the test files committed, before dev writes production 

210 # code). The orchestrator's dispatch table reads this flag when 

211 # state==TESTS_RED to decide whether to fire ``harness_precheck`` 

212 # or skip ahead to ``dev``. Default ``False`` so existing stories 

213 # transparently get the precheck on their next visit. 

214 harness_precheck_passed: bool = False 

215 # Phase 3 EBS (Evidence-Based Scheduling). PM assigns Fibonacci 

216 # story points at split time (1, 2, 3, 5, 8, 13). The chain 

217 # computes ``estimated_seconds`` from per-(persona, points) 

218 # baselines at story-creation time; the Monte Carlo simulator 

219 # uses points + persona velocities to project per-direction 

220 # ETAs. Both columns are nullable — legacy stories spawned 

221 # before this column existed keep ``None``. 

222 points: int | None = None 

223 estimated_seconds: float | None = None 

224 # WS1.2 independent acceptance oracle. Path (relative to the factory root) 

225 # of the acceptance test authored from THIS story's direction acceptance 

226 # criteria — the SPEC ONLY, blind to the dev's implementation and the dev's 

227 # own tests. Authored EARLY (at story spawn, before dev runs) and stored in 

228 # factory state UNDER ``state/acceptance/<app>/<story_id>/`` — deliberately 

229 # outside the app repo and outside the per-story dev worktree, so the dev 

230 # sandbox never receives it (the anti-reward-hack property: a coder that 

231 # cannot see the test judging it cannot special-case it — ImpossibleBench). 

232 # None means no oracle FILE is stored yet — which is NOT the same as "not 

233 # required": authoring can flake (transient LLM error) for a story that DOES 

234 # need an oracle. The required/blocking decision keys off 

235 # ``acceptance_expected`` below, never off this ref, so a failed author 

236 # cannot silently ship un-gated code. 

237 acceptance_test_ref: str | None = None 

238 # WS1.2 — whether this story is EXPECTED to have an acceptance oracle, set at 

239 # spawn to ``bool(gates.acceptance_oracle and direction.acceptance)``, 

240 # INDEPENDENT of whether authoring succeeded. This is the single source of 

241 # truth for "acceptance-verified is required for this story". When True but 

242 # ``acceptance_test_ref`` is missing (authoring flaked), the gate BLOCKS 

243 # authoritatively and the tick self-heal re-authors from the spec on a later 

244 # tick — expected-but-missing never silently passes and never blocks forever. 

245 acceptance_expected: bool = False

406 (StoryState.DOCS_ONBOARDER_DONE, EVENT_BUDGET_EXCEEDED): StoryState.BLOCKED_BUDGET_EXCEEDED, 

407 (StoryState.DEPLOY_PENDING, EVENT_DEPLOY_STARTED): StoryState.DEPLOY_PENDING,

414class IllegalTransitionError(ValueError): 

415 """Raised when ``advance`` is called with a (state, event) pair not in 

416 the transition table."""

419def advance(story: StoryRecord, event: str, payload: dict[str, Any] | None = None) -> StoryState: 

420 """Compute the next state for ``story`` given ``event`` (and optional ``payload``). 

421 

422 Pure: this function does not mutate ``story``, does not touch the 

423 database, does not call any handler. The caller is responsible for 

424 persisting the new state. Returns the ``StoryState`` enum value. 

425 

426 Special-case logic: 

427 * ``EVENT_DEV_TESTS_RED`` advances to ``DEV_RETRY``. The caller MUST 

428 also bump ``story.dev_retries`` and may bump ``current_model_tier`` 

429 before invoking the dev handler again. When retries exhaust, the 

430 caller passes ``EVENT_DEV_EXHAUSTED`` instead. 

431 """ 

432 current = StoryState(story.state) 

433 key = (current, event) 

434 if key not in _TRANSITIONS: 

435 raise IllegalTransitionError(f"No transition: state={current.value!r} event={event!r}") 

436 return _TRANSITIONS[key]

439def is_terminal(state: StoryState) -> bool: 

440 """A state is terminal for the chain when no outgoing transitions exist.""" 

441 return not any(s == state for (s, _) in _TRANSITIONS)

444def list_transitions_from(state: StoryState) -> list[tuple[str, StoryState]]: 

445 """Return ``[(event, next_state)]`` for all transitions out of ``state``. 

446 

447 Useful for the `factory story <id>` CLI and for tests. 

448 """ 

449 out: list[tuple[str, StoryState]] = [] 

450 for (s, ev), ns in _TRANSITIONS.items(): 

451 if s == state: 

452 out.append((ev, ns)) 

453 return out
