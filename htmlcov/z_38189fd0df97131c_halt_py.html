1"""factory.manager.halt — Halt authority for the factory (Phase 7). 

2 

3Only the L3 Diagnostician may *request* a halt (via ``request_halt``). 

4Only a human operator may *clear* a halt (via ``clear_halt``). 

5The driver loop and ``tick()`` each call ``is_halted`` before dispatching. 

6 

7OPERATOR-ONLY MODULE 

8-------------------- 

9This file MUST NOT be invoked by any LLM pathway. It is in the 

10``factory/manager/*.py`` forbidden class and will never be auto-applied 

11by the L4 pipeline. ``clear_halt`` in particular must only be called by 

12``factory resume`` (a CLI command requiring human interaction). 

13""" 

14 

15from __future__ import annotations 

16 

17import json 

18import time 

19from datetime import UTC, datetime 

20from pathlib import Path # noqa: E402 

21 

22# Schema version for the halt state file. 

23_SCHEMA_VERSION = 1 

24 

25# Bounded retry for reading the halt file. A halt exists precisely to STOP the 

26# factory, so a read/parse failure must not silently be treated as "not 

27# halted". We retry a few times with a tiny backoff to ride out a transient FS 

28# glitch (a half-written file mid-flush, a momentary EIO) before concluding. 

29_HALT_READ_RETRIES = 3 

30_HALT_READ_BACKOFF_S = 0.05 

31 

32# Standard filename for the halt mode state. 

33_HALT_FILE = "factory_mode.json" 

34 

35# Standard filename for the halt history archive. 

36_HALT_HISTORY_FILE = ".halt_history.json" 

37 

38# Grace window after an operator clears a halt during which the manager may 

39# NOT re-halt. An operator resume is an explicit override; stall-class 

40# concerns (e.g. "no ticks for N minutes") can only clear AFTER the resume 

41# lets the orchestrator run again, so an immediate re-halt deadlocks the 

42# factory against its own manager: halt blocks ticks -> tick gap grows -> 

43# re-halt. Observed live 2026-06-11: L3 re-halted 94 seconds after an 

44# operator resume, before the first post-resume tick could land. 

45_RESUME_GRACE_MINUTES = 30 

46 

47 

48def _halt_path(root: Path) -> Path: 

49 return root / "state" / _HALT_FILE 

50 

51 

52def _history_path(root: Path) -> Path: 

53 return root / "state" / _HALT_HISTORY_FILE

56def request_halt( 

57 *, 

58 root: Path, 

59 concern_title: str, 

60 proposal_path: str | None, 

61 reason: str, 

62) -> Path | None: 

63 """Write the halt mode file. 

64 

65 Idempotent — if halt is already set, append the old state to the 

66 history archive and overwrite with the new halt (most recent wins). 

67 

68 Returns the path to the halt state file, or ``None`` when the halt was 

69 suppressed because an operator cleared a halt less than 

70 ``_RESUME_GRACE_MINUTES`` ago (the factory gets that window to 

71 demonstrate liveness before the manager may halt it again). 

72 

73 Parameters 

74 ---------- 

75 root: 

76 Factory root directory. 

77 concern_title: 

78 The L3 concern title that triggered this halt request. 

79 proposal_path: 

80 Absolute path (as str) to the proposal file that carried the 

81 halt request. May be None when called from tests or tooling. 

82 reason: 

83 Free-text justification from the L3 proposal's ``halt_reason`` 

84 field. 

85 """ 

86 root = Path(root) 

87 halt_path = _halt_path(root) 

88 

89 # Operator-resume grace: a recent manual clear overrides halt authority. 

90 last_cleared = _last_operator_clear_at(root) 

91 if last_cleared is not None: 

92 age_s = (datetime.now(UTC) - last_cleared).total_seconds() 

93 if age_s < _RESUME_GRACE_MINUTES * 60: 

94 return None 

95 

96 # If halt already set, archive the previous state before overwriting. 

97 if halt_path.exists(): 

98 try: 

99 old_state = json.loads(halt_path.read_text(encoding="utf-8")) 

100 _append_history(root, old_state) 

101 except (OSError, json.JSONDecodeError): 

102 pass 

103 

104 state: dict = { 

105 "schema_version": _SCHEMA_VERSION, 

106 "mode": "halted", 

107 "set_at": datetime.now(UTC).isoformat(), 

108 "set_by": "manager_diagnostician", 

109 "concern_title": concern_title, 

110 "proposal_path": proposal_path, 

111 "reason": reason, 

112 } 

113 

114 halt_path.parent.mkdir(parents=True, exist_ok=True) 

115 halt_path.write_text(json.dumps(state, indent=2), encoding="utf-8") 

116 return halt_path

119def is_halted(*, root: Path) -> bool: 

120 """Return True if the factory should be treated as halted. 

121 

122 Fail-SAFE semantics 

123 ------------------- 

124 * No halt file → not halted (normal running state). 

125 * Valid file, mode set → honour ``mode == "halted"``. 

126 * File present but the contents cannot be read/parsed after a bounded 

127 retry → **treat as halted** and emit a CRITICAL alert. 

128 

129 The last rule is the important one. Historically this path returned 

130 ``False`` (fail-OPEN): a corrupt/unreadable halt file made the factory 

131 treat itself as NOT halted and keep dispatching — silently ignoring the 

132 very halt that was meant to stop it. That is the single most dangerous 

133 failure this module can have. 

134 

135 Tradeoff: a *persistent* read error over-halts the factory. That is 

136 acceptable because a visible over-halt is fully recoverable — an operator 

137 sees the alert and clears it with ``factory resume`` (``clear_halt`` is 

138 tolerant of a corrupt file for exactly this reason) — whereas a silently 

139 ignored halt is not recoverable and defeats the control plane. The bounded 

140 retry keeps a mere transient FS blip from wedging the factory. 

141 """ 

142 root = Path(root) 

143 p = _halt_path(root) 

144 if not p.exists(): 

145 return False 

146 last_exc: Exception | None = None 

147 for attempt in range(_HALT_READ_RETRIES): 

148 try: 

149 state = json.loads(p.read_text(encoding="utf-8")) 

150 return state.get("mode") == "halted" 

151 except (OSError, json.JSONDecodeError) as exc: 

152 last_exc = exc 

153 if attempt < _HALT_READ_RETRIES - 1: 

154 time.sleep(_HALT_READ_BACKOFF_S) 

155 # Every read failed. Fail SAFE (halted) and make it unmistakable. 

156 _alert_halt_unreadable(root, p, last_exc) 

157 return True

160def get_halt_state(*, root: Path) -> dict | None: 

161 """Return the halt state dict or None if not halted. 

162 

163 Uses the same bounded retry as :func:`is_halted`. Returns ``None`` only 

164 when the file is genuinely absent or a valid file has a non-halted mode. 

165 If the file is present but unreadable after retries, returns a synthetic 

166 ``mode="halted"`` state so callers that display the halt reason stay 

167 consistent with :func:`is_halted` (which fail-safes to halted). 

168 """ 

169 root = Path(root) 

170 p = _halt_path(root) 

171 if not p.exists(): 

172 return None 

173 last_exc: Exception | None = None 

174 for attempt in range(_HALT_READ_RETRIES): 

175 try: 

176 state = json.loads(p.read_text(encoding="utf-8")) 

177 if state.get("mode") == "halted": 

178 return state 

179 return None 

180 except (OSError, json.JSONDecodeError) as exc: 

181 last_exc = exc 

182 if attempt < _HALT_READ_RETRIES - 1: 

183 time.sleep(_HALT_READ_BACKOFF_S) 

184 _alert_halt_unreadable(root, p, last_exc) 

185 return { 

186 "mode": "halted", 

187 "reason": "halt file present but unreadable/corrupt (fail-safe halt)", 

188 "set_by": "fail_safe", 

189 }

192def clear_halt( 

193 *, 

194 root: Path, 

195 cleared_by: str = "operator", 

196 reason: str | None = None, 

197) -> dict: 

198 """Clear the halt mode. 

199 

200 Moves the current halt state to ``state/.halt_history.json`` and 

201 removes (or resets) ``state/factory_mode.json``. 

202 

203 OPERATOR-ONLY — must never be called by any LLM path. 

204 

205 Parameters 

206 ---------- 

207 root: 

208 Factory root directory. 

209 cleared_by: 

210 Who cleared the halt. Defaults to "operator". 

211 reason: 

212 Optional free-text reason for clearing. 

213 

214 Returns 

215 ------- 

216 dict 

217 The archived halt state (with ``cleared_at``, ``cleared_by``, 

218 ``clear_reason`` added). 

219 

220 Raises 

221 ------ 

222 FileNotFoundError 

223 If there is no active halt to clear. 

224 """ 

225 root = Path(root) 

226 halt_path = _halt_path(root) 

227 

228 if not halt_path.exists(): 

229 raise FileNotFoundError( 

230 f"No halt state file found at {halt_path}; nothing to clear." 

231 ) 

232 

233 # Read the current halt state, tolerating a corrupt/unreadable file. This 

234 # is essential: is_halted() fail-safes to *halted* on a corrupt file, so an 

235 # operator MUST still be able to clear that halt with `factory resume`. If 

236 # we raised here on a parse error, a corrupt halt file would wedge the 

237 # factory permanently — the exact deadlock the fail-safe posture must avoid. 

238 try: 

239 state = json.loads(halt_path.read_text(encoding="utf-8")) 

240 except (OSError, json.JSONDecodeError) as exc: 

241 state = { 

242 "mode": "halted", 

243 "reason": "halt file unreadable/corrupt at clear time", 

244 "corrupt_read_error": repr(exc), 

245 } 

246 if not isinstance(state, dict): 

247 state = {"mode": "halted", "reason": "halt file had non-object contents"} 

248 if state.get("mode") != "halted": 

249 raise ValueError( 

250 f"factory_mode.json exists but mode={state.get('mode')!r} (not 'halted'); " 

251 "nothing to clear." 

252 ) 

253 

254 # Annotate the archived entry with clearance metadata. 

255 archived = dict(state) 

256 archived["cleared_at"] = datetime.now(UTC).isoformat() 

257 archived["cleared_by"] = cleared_by 

258 if reason is not None: 

259 archived["clear_reason"] = reason 

260 

261 _append_history(root, archived) 

262 

263 # Remove the halt file so is_halted() returns False. 

264 halt_path.unlink() 

265 

266 return archived

274def _alert_halt_unreadable(root: Path, path: Path, exc: Exception | None) -> None: 

275 """Emit a CRITICAL, visible alert that the halt file could not be read. 

276 

277 Best-effort: alerting must never itself raise out of the halt check. The 

278 fail-safe decision (treat as halted) does not depend on this succeeding. 

279 """ 

280 try: 

281 from factory.manager.signals import write_alert_event 

282 

283 write_alert_event( 

284 "halt_unreadable", 

285 f"halt file present but unreadable/corrupt after " 

286 f"{_HALT_READ_RETRIES} attempts; failing SAFE (treating factory as " 

287 f"HALTED). Clear with `factory resume` once the file is fixed.", 

288 severity="critical", 

289 software_factory_root=root, 

290 halt_path=str(path), 

291 read_error=repr(exc) if exc is not None else None, 

292 ) 

293 except Exception: # noqa: BLE001 - alerting is best-effort; never raise here 

294 import sys as _sys 

295 

296 print( 

297 f"[halt] CRITICAL: halt file {path} unreadable and alert emit failed: {exc!r}", 

298 file=_sys.stderr, 

299 )

302def _last_operator_clear_at(root: Path) -> datetime | None: 

303 """Return when an operator last cleared a halt, or None. 

304 

305 Scans the history archive newest-first for an entry carrying 

306 ``cleared_at`` (only ``clear_halt`` writes that field; entries archived 

307 by ``request_halt`` overwrites do not). 

308 """ 

309 history_path = _history_path(root) 

310 if not history_path.exists(): 

311 return None 

312 try: 

313 data = json.loads(history_path.read_text(encoding="utf-8")) 

314 except (OSError, json.JSONDecodeError): 

315 return None 

316 if not isinstance(data, list): 

317 return None 

318 for entry in reversed(data): 

319 if isinstance(entry, dict) and entry.get("cleared_at"): 

320 try: 

321 return datetime.fromisoformat(entry["cleared_at"]) 

322 except (TypeError, ValueError): 

323 return None 

324 return None

327def _append_history(root: Path, entry: dict) -> None: 

328 """Append ``entry`` to the halt history archive.""" 

329 history_path = _history_path(root) 

330 history: list[dict] = [] 

331 if history_path.exists(): 

332 try: 

333 data = json.loads(history_path.read_text(encoding="utf-8")) 

334 if isinstance(data, list): 

335 history = data 

336 except (OSError, json.JSONDecodeError): 

337 pass 

338 history.append(entry) 

339 history_path.parent.mkdir(parents=True, exist_ok=True) 

340 history_path.write_text(json.dumps(history, indent=2), encoding="utf-8")
