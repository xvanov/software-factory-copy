1"""Detector: state_distribution_skew — surface uneven story-state distributions. 

2 

3This module exposes ``state_distribution_skew``, which reads the most 

4recent ``queue_snapshot`` event per app from ``state/events/queue.ndjson`` 

5and returns per-app state-distribution observations including fractions 

6and a threshold flag. The calling agent decides whether an observed 

7skew warrants action. 

8""" 

9 

10from __future__ import annotations 

11 

12import json 

13from datetime import datetime 

14from pathlib import Path 

15 

16 

17def state_distribution_skew( 

18 *, 

19 root: Path, 

20 since: datetime, 

21 threshold_fraction: float = 0.5, 

22) -> dict: 

23 """Return state-distribution observations per app from recent queue snapshots. 

24 

25 Reads ``state/events/queue.ndjson`` and finds the *most recent* 

26 ``queue_snapshot`` event per app that occurred on or after *since*. 

27 For each app, computes the fraction of stories in each state and 

28 flags when any single state accounts for more than 

29 *threshold_fraction* of the total. 

30 

31 Parameters 

32 ---------- 

33 root: 

34 Factory root directory. 

35 since: 

36 Lower bound (inclusive) on the snapshot's ``ts`` field. 

37 Snapshots older than *since* are ignored. 

38 threshold_fraction: 

39 Fraction above which ``exceeds_threshold`` is set to ``True``. 

40 Default 0.5 (50 %). This is an *observation* label, not a 

41 decision — the calling agent interprets its significance. 

42 

43 Returns 

44 ------- 

45 dict 

46 ``{"app_snapshots": {<app>: {...}, ...}}`` 

47 

48 Each app value contains: 

49 

50 * ``ts`` — ISO-8601 timestamp of the snapshot 

51 * ``counts_by_state`` — raw state→count mapping 

52 * ``total`` — sum of all counts 

53 * ``max_state`` — state with the highest count (empty string 

54 when total is 0) 

55 * ``max_fraction`` — fraction of total occupied by max_state 

56 (0.0 when total is 0) 

57 * ``exceeds_threshold`` — True when max_fraction > 

58 threshold_fraction 

59 * ``exceeds_state`` — the name of the state that exceeds the 

60 threshold, or None 

61 

62 Returns ``{"app_snapshots": {}}`` when no qualifying snapshots 

63 exist. 

64 """ 

65 stream = root / "state" / "events" / "queue.ndjson" 

66 if not stream.exists(): 

67 return {"app_snapshots": {}} 

68 

69 since_iso = since.isoformat() 

70 # Keep only the most recent snapshot per app 

71 latest: dict[str, dict] = {} 

72 

73 with stream.open(encoding="utf-8") as fh: 

74 for raw in fh: 

75 raw = raw.strip() 

76 if not raw: 

77 continue 

78 try: 

79 rec = json.loads(raw) 

80 except json.JSONDecodeError: 

81 continue 

82 if rec.get("event") != "queue_snapshot": 

83 continue 

84 ts = rec.get("ts", "") 

85 if ts < since_iso: 

86 continue 

87 app = rec.get("app", "") 

88 if not app: 

89 continue 

90 if app not in latest or ts > latest[app].get("ts", ""): 

91 latest[app] = rec 

92 

93 app_snapshots: dict[str, dict] = {} 

94 for app, rec in latest.items(): 

95 counts: dict[str, int] = rec.get("counts_by_state") or {} 

96 total = sum(counts.values()) 

97 if total > 0: 

98 max_state = max(counts, key=lambda s: counts[s]) 

99 max_fraction = counts[max_state] / total 

100 else: 

101 max_state = "" 

102 max_fraction = 0.0 

103 exceeds = max_fraction > threshold_fraction 

104 app_snapshots[app] = { 

105 "ts": rec.get("ts", ""), 

106 "counts_by_state": counts, 

107 "total": total, 

108 "max_state": max_state, 

109 "max_fraction": max_fraction, 

110 "exceeds_threshold": exceeds, 

111 "exceeds_state": max_state if exceeds else None, 

112 } 

113 

114 return {"app_snapshots": app_snapshots}
